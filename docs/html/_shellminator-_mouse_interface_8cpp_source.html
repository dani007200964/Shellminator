58void Shellminator::pushMouseEvent( uint8_t x, uint8_t y, uint8_t event ){
59
60 // This solution uses a circular buffer, so it has two pointers.
61 // The write pointer always tracks the next free slot in the buffer.
62 // Store the new values there.
63 mouseBuffer[ mouseBufferWritePtr ].x = x;
64 mouseBuffer[ mouseBufferWritePtr ].y = y;
65 mouseBuffer[ mouseBufferWritePtr ].event = event;
66
67 // Increment the write pointer.
68 mouseBufferWritePtr++;
69
70 // Detect overflow.
71 if( mouseBufferWritePtr >= MOUSE_BUFFER_SIZE ){
72 // Handle overflow.
73 mouseBufferWritePtr = 0;
74 }
75
76}

78int Shellminator::mouseAvailable(){
79
80 // This solution uses a circular buffer, so it has two pointers.
81 // One for read and one for writing.
82 // If they point to the same place, that means, there is no data available.
83 if( mouseBufferWritePtr == mouseBufferReadPtr ){
84 return 0;
85 }
86
87 // The first case is, when the Write pointer is ahead of the read pointer.
88 else if( mouseBufferWritePtr > mouseBufferReadPtr ){
89 return mouseBufferWritePtr - mouseBufferReadPtr;
90 }
91
92 // The second case is, when the Read pointer is 'ahead' of the write pointer.
93 // It happens when overflow event occurs with the circular buffer.
94 else{
95
96 return MOUSE_BUFFER_SIZE - mouseBufferReadPtr + mouseBufferWritePtr;
97
98 }
99
100}

102Shellminator::mouseEvent_t Shellminator::mouseRead(){
103
104 // This struct will be returned.
105 mouseEvent_t ret;
106
107 // By default set the returned event to invalid.
108 ret.event = MOUSE_INVALID;
109
110 // This solution uses a circular buffer, so it has two pointers.
111 // One for read and one for writing.
112 // If they point to the same place, that means, there is no data available.
113 if( mouseBufferWritePtr == mouseBufferReadPtr ){
114
115 // This case return with invalid event code.
116 return ret;
117
118 }
119
120 // Any other case, that means it has at least one new item to be read.
121 else{
122
123 // Get the next data from the circular structure.
124 ret = mouseBuffer[ mouseBufferReadPtr ];
125
126 // Increment the read pointer.
127 mouseBufferReadPtr++;
128
129 // Check for overflow.
130 if( mouseBufferReadPtr >= MOUSE_BUFFER_SIZE ){
131
132 // Handle overflow.
133 mouseBufferReadPtr = 0;
134 }
135
136 }
137
138 // Return with the next element data.
139 return ret;
140
141}

143void Shellminator::ShellminatorMouseEventParserState( char new_char ){
144
145 // Replace ';' to ' '. It will be easier to parse later.
146 if( new_char == ';' ){
147 new_char = ' ';
148 }
149
150 // Save the new character to the buffer and increment the pointer.
151 mouseEventBuffer[ mouseEventBufferCntr ] = new_char;
152 mouseEventBufferCntr++;
153
154 // Detect if the terminatong m or M character received.
155 if( ( new_char == 'm' ) || ( new_char == 'M' ) ){
156
157 parseMouseData();
158
159 // Reset the pointer.
160 mouseEventBufferCntr = 0;
161
162 // Reset to the default state.
163 currentState = &Shellminator::ShellminatorDefaultState;
164 return;
165
166 }
167
168 // Buffer overflow detection.
169 if( mouseEventBufferCntr >= SHELLMINATOR_MOUSE_PARSER_BUFFER_SIZE ){
170
171 // Reset the pointer.
172 mouseEventBufferCntr = 0;
173
174 // Reset to the default state.
175 currentState = &Shellminator::ShellminatorDefaultState;
176
177 }
178
179}

181void Shellminator::parseMouseData(){
182
183 int xPos;
184 int yPos;
185 int event;
186
187 bool release;
188
189 // Overflow protection.
190 if( mouseEventBufferCntr < 1 ){
191 return;
192 }
193
194 // Detect if the button is pressed or released.
195 release = mouseEventBuffer[ mouseEventBufferCntr - 1 ] == 'm';
196
197 // Replace the end character with string terminator.
198 mouseEventBuffer[ mouseEventBufferCntr - 1 ] = '\0';
199
200 sscanf( mouseEventBuffer, "%d %d %d", &event, &xPos, &yPos );
201
202 switch( event ){
203
204 // Left mouse.
205 case 0:
206 event = release ? MOUSE_LEFT_RELEASED : MOUSE_LEFT_PRESSED;
207 break;
208
209 // Middle mouse.
210 case 1:
211 event = release ? MOUSE_MIDDLE_RELEASED : MOUSE_MIDDLE_PRESSED;
212 break;
213
214 // Right mouse.
215 case 2:
216 event = release ? MOUSE_RIGHT_RELEASED : MOUSE_RIGHT_PRESSED;
217 break;
218
219 // Mouse wheel up.
220 case 64:
221 event = MOUSE_WHEEL_UP;
222 break;
223
224 // Mouse wheel down.
225 case 65:
226 event = MOUSE_WHEEL_DOWN;
227 break;
228
229 // Any other means invalid.
230 default:
231 event = MOUSE_INVALID;
232 break;
233
234 }
235
236 pushMouseEvent( xPos, yPos, event );
237
238}
