68void ShellminatorList::update( int width_p, int height_p ){
69
70 Shellminator::shellEvent_t newEvent;
71
72 if( parent == NULL ){
73 return;
74 }
75
76 width = width_p;
77 height = height_p;
78
79 if( height > ( listSize + 1 ) ){
80 height = listSize + 1;
81 }
82
83 newEvent = parent -> readEvent();
84
85 // In case of empty event, we can't do anything.
86 if( newEvent.type == Shellminator::SHELL_EVENT_EMPTY ){
87 return;
88 }
89
90 if( newEvent.type == Shellminator::SHELL_EVENT_RESIZE ){
91 redraw = true;
92 return;
93 }
94
95 if( newEvent.type == Shellminator::SHELL_EVENT_CODED_KEY ){
96
97 if( newEvent.eventCode == Shellminator::EVENT_CODE_UP_ARROW ){
98 selected--;
99 if( ( selected < drawOffset ) && ( drawOffset > 0 ) ){
100 drawOffset--;
101 }
102 if( selected < 0 ){
103 selected = 0;
104 }
105 redraw = true;
106 parent -> requestRedraw();
107 return;
108 }
109
110 if( newEvent.eventCode == Shellminator::EVENT_CODE_DOWN_ARROW ){
111 selected++;
112 if( ( selected > ( height - 2 + drawOffset ) ) && ( drawOffset < ( listSize - ( height - 1 ) ) ) ){
113 drawOffset++;
114 }
115 if( selected >= listSize ){
116 selected = listSize - 1;
117 }
118 redraw = true;
119 parent -> requestRedraw();
120 return;
121 }
122
123 if( newEvent.eventCode == Shellminator::EVENT_CODE_RETURN ){
124 if( func == NULL ){
125 return;
126 }
127 func( optionsList, listSize, selected, this );
128 return;
129 }
130
131 }

146void ShellminatorList::draw( bool noClear ){
147
148 // Generic counter.
149 int i;
150
151 // Used to safe array indexing.
152 int index;
153
154 // Check for the necessary parameters.
155 if( parent == NULL ){
156 return;
157 }
158
159 if( channel == NULL){
160 return;
161 }
162
163 if( optionsList == NULL ){
164 return;
165 }
166
167 if( listSize == 0 ){
168 return;
169 }
170
171 if( instruction == NULL ){
172 return;
173 }
174
175 // Only draw if resized event or timer event happened.
176 if( !redraw ){
177 return;
178 }
179
180 redraw = false;
181
182 parent -> format( channel, Shellminator::REGULAR, Shellminator::WHITE );
183
184 // Set cursor to top left and print the instruction text.
185 Shellminator::setCursorPosition( channel, originX, originY );
186
187 i = 0;
188 instructionLines = 0;
189 while( instruction[ i ] ){
190 if( instruction[ i ] == '\n' ){
191 instructionLines++;
192 Shellminator::setCursorPosition( channel, originX, originY + instructionLines );
193 if( !noClear ){
194 channel -> print( "\033[0K" );
195 }
196 }
197 else{
198 channel -> print( instruction[ i ] );
199 }
200 i++;
201 }
202
203 // Print as many elements as high the remaining screen is.
204 for( i = 1; i < height; i++ ){
205
206 // Calculate the index of the current option.
207 index = ( i - 1 ) + drawOffset;
208
209 // Set cursor to the current options location.
210 Shellminator::setCursorPosition( channel, originX, originY + i + instructionLines );
211
212 // Check if the current option is the selected one.
213 if( index == selected ){
214 // If selected change the format to
215 // background to highlight the selection.
216 // Also print some graphics to the beginning.
217 parent -> format( channel, Shellminator::BACKGROUND, Shellminator::WHITE );
218 channel -> print( "\u2BA9 " );
219 }
220
221 else{
222 // Otherwise, just print blank spaces.
223 channel -> print( " " );
224 }
225
226 // If we are not at the top of the list,
227 // we have to indicate that, we have more
228 // options 'hidden' upwards.
229 if( ( i == 1 ) && ( drawOffset > 0 ) ){
230 channel -> print( "\u2191 " );
231 }
232
233 // If we are not at the bottom of the list,
234 // we have to indicate that, we have more
235 // options 'hidden' downwards.
236 else if( ( i == ( height - 1 ) ) && ( drawOffset < ( listSize - ( height - 1 ) ) ) ){
237 channel -> print( "\u2193 " );
238 }
239
240 // Otherwise, we just print blank spaces.
241 else{
242 channel -> print( " " );
243 }
244
245 // Print the option with a blak space at the end.
246 channel -> print( optionsList[ index ] );
247 channel -> print( ' ' );
248
249 // Check if the current option is the selected one.
250 if( index == selected ){
251 // If selected, we have to change back the formatting to regular.
252 parent -> format( channel, Shellminator::REGULAR, Shellminator::WHITE );
253 }
254
255 printExtra( index, noClear );
256
257 }
258
259}
