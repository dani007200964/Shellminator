103 void ShellminatorWebServer::update(){
104 char newChar;
105 bool urlProcessed = false;
106 int endOfURL;
107
108 if( server == NULL ){
109 return;
110 }
111
112 client = server -> accept();
113
114 if( client ){
115 SHELLMINATOR_WEBS_DBGLN( __CONST_TXT__( "New Web Server Client." ) );
116 clientBuffer[ 0 ] = '\0';
117 clientBufferCounter = 0;
118 timerStart = millis();
119
120 while( CLIENT_STATE ){
121 if( ( millis() - timerStart ) > SHELLMINATOR_WEBSERVER_CLIENT_TIMEOUT_MS ){
122 SHELLMINATOR_WEBS_DBGLN( __CONST_TXT__( "Timeout!" ) );
123 break;
124 }
125
126 if( client.available() ){
127 newChar = client.read();
128
129 //SHELLMINATOR_WEBS_DBG( (char)newChar );
130 //SHELLMINATOR_WEBS_DBG( __CONST_TXT__( "\t0x" ) );
131 //SHELLMINATOR_WEBS_DBG( __CONST_TXT__( "\t" ) );
132 //SHELLMINATOR_WEBS_DBGLN( (int)newChar );
133
134 if( newChar == '\n' ){
135 clientBuffer[ clientBufferCounter ] = '\0';
136 clientBufferCounter = 0;
137
138 if( clientBuffer[ 0 ] == '\0' ){
139 SHELLMINATOR_WEBS_DBGLN( __CONST_TXT__( "Header end detected." ) );
140 break;
141 }
142
143 else if( urlProcessed ){
144 SHELLMINATOR_WEBS_DBG( __CONST_TXT__( "Client Data: " ) );
145 SHELLMINATOR_WEBS_DBGLN( __CONST_TXT__( clientBuffer ) );
146 continue;
147 }
148
149 else if( startsWith( clientBuffer, "GET " ) ){
150 endOfURL = indexOf( clientBuffer + 4, ' ' );
151 if( endOfURL >= 0 ){
152 clientBuffer[ endOfURL + 4 ] = '\0';
153
154 // We have to check for a '?' character as well.
155 // This character indicates that we have arguments in the
156 // URL and we do not want to pass them to the callbacks.
157 endOfURL = indexOf( clientBuffer, '?' );
158 if( endOfURL >= 0 ){
159 clientBuffer[ endOfURL ] = '\0';
160 }
161
162 shiftStringLeft( clientBuffer );
163 shiftStringLeft( clientBuffer );
164 shiftStringLeft( clientBuffer );
165 shiftStringLeft( clientBuffer );
166
167 processURL( clientBuffer );
168 urlProcessed = true;
169 }
170 }
171
172 }
173
174 else if( newChar == '\r' ){
175 continue;
176 }
177
178 else{
179 clientBuffer[ clientBufferCounter ] = newChar;
180 clientBufferCounter++;
181 if( clientBufferCounter >= SHELLMINATOR_WEBSERVER_URL_BUFFER_SIZE ){
182 clientBufferCounter = SHELLMINATOR_WEBSERVER_URL_BUFFER_SIZE;
183 }
184 }
185 }
186 }
187 client.stop();
188 SHELLMINATOR_WEBS_DBGLN( __CONST_TXT__( "Client Disconnected!" ) );
189 }
190
191 }
