36void Shellminator::input( char *buffer, int bufferSize, const char *instruction, void(*callback)(char*, int, Shellminator*), bool secret ){
37
38 // Create a new line for the input prompt.
39 channel -> println();
40
41 // Save the buffer and its size.
42 inputDestinationBuffer = buffer;
43 inputDestinationBufferSize = bufferSize;
44
45 // If the output buffer is larger, than the internal buffer,
46 // limit its size. The terminal interface can't handle more
47 // characters anyway.
48 if( inputDestinationBufferSize > SHELLMINATOR_BUFF_LEN ){
49 inputDestinationBufferSize = SHELLMINATOR_BUFF_LEN;
50 }
51
52 // Save the callback function address.
53 inputCallback = callback;
54
55 // Calculate the text size for the instruction text.
56 inputInstuctionSize = strlen( instruction );
57
58 // Print the instruction.
59 channel -> print( instruction );
60
61 // Save the flag for the input prompt.
62 inputActive = true;
63
64 // Save the secret mode setting.
65 inputSecretMode = secret;
66
67 // In case of secret mode, print a lock
68 // and midify the instruction text size
69 // accordingly.
70 if( inputSecretMode ){
71 channel -> print( "\U0001F512 " );
72 inputInstuctionSize += 3;
73 }
74
75 // Redraw the line.
76 redrawLine();
77}
