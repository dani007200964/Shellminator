97bool ShellminatorWebSocket::sendFrame( ShellminatorWebSocket::wsDecodedHeaderType_t type, const uint8_t* data, uint32_t dataSize ){
98
99 int i;
100 bool ret;
101
102 if( data == NULL ){
103 return sendFrame125( type, data, dataSize );
104 }
105
106 for( i = 0; i < (int)( dataSize / 125 ); i++ ){
107 ret = sendFrame125( type, data, 125 );
108 if( !ret ){
109 return false;
110 }
111 data += 125;
112 }
113
114 return sendFrame125( type, data, dataSize % 125 );
115}

117bool ShellminatorWebSocket::sendFrame125( ShellminatorWebSocket::wsDecodedHeaderType_t type, const uint8_t* data, uint8_t dataSize ){
118 int i;
119 uint8_t maskBuffer[ 4 ] = { 0 };
120 uint8_t buffer[ WS_FRAME_HEADER_SIZE ] = { 0 };
121
122 uint8_t headerSize = 2;
123
124 if( dataSize > 125 ){
125 return false;
126 }
127
128 // Byte 0
129 buffer[ 0 ] = 0;
130 buffer[ 0 ] |= 1UL << 7; // To set the Fin bit.
131 buffer[ 0 ] |= type;
132
133 // Byte 1
134 buffer[ 1 ] = 0x00;
135 buffer[ 1 ] |= dataSize;
136
137 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "---- Header To Send ----" ) );
138 for( i = 0; i < headerSize; i++ ){
139 SHELLMINATOR_WS_DBG( i );
140 SHELLMINATOR_WS_DBG( __CONST_TXT__( "\t0x" ) );
141 SHELLMINATOR_WS_DBG( buffer[ i ], HEX );
142 SHELLMINATOR_WS_DBG( __CONST_TXT__( "\t" ) );
143 SHELLMINATOR_WS_DBGLN( (int)buffer[ i ] );
144 }
145 client.write( (const uint8_t*)buffer, headerSize );
146
147 if( dataSize > 0 ){
148 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "---- Data To Send ----" ) );
149 for( i = 0; i < dataSize; i++ ){
150 SHELLMINATOR_WS_DBG( i );
151 SHELLMINATOR_WS_DBG( __CONST_TXT__( "\t0x" ) );
152 SHELLMINATOR_WS_DBG( data[ i ], HEX );
153 SHELLMINATOR_WS_DBG( __CONST_TXT__( "\t" ) );
154 SHELLMINATOR_WS_DBGLN( (int)data[ i ] );
155 }
156 client.write( (const uint8_t*)data, dataSize );
157 }
158 return true;
159}

161void ShellminatorWebSocket::update(){
162 char newChar;
163 WiFiClient newClient;
164
165 if( server == NULL ){
166 return;
167 }
168
169 newClient = server -> accept();
170
171 if( newClient ){
172 if( CLIENT_STATE ){
173 // Reject a new client, because another one is alredy in use.
174 newClient.stop();
175 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "WS Client already in use. Rejecting new connection." ) );
176 }
177 else{
178 // New connection.
179 client = newClient;
180 #ifndef ARDUINO_UNOWIFIR4
181 client.setNoDelay( true );
182 #endif
183 clientConnected = true;
184 wsState = WS_HEADER_STATE;
185 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "New WS Client." ) );
186 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "---- Processing Client Header ----" ) );
187 }
188 }
189
190 // Check for disconnection event
191 if( clientConnected && !CLIENT_STATE ){
192 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "Disconnect Event Detected!" ) );
193 closeClient();
194 }
195
196 // Process webscoket data.
197 if( clientConnected && CLIENT_STATE ){
198
199 while( client.available() ){
200 newChar = client.read();
201 if( wsState == WS_HEADER_STATE ){
202 wsHeaderProcessing( newChar );
203 }
204 else if( wsState == WS_CONNECTED_STATE ){
205 wsDataProcessing( newChar );
206 }
207 }
208
209 }

272void ShellminatorWebSocket::wsDataProcessing( char newChar ){
273
274 uint8_t currentPayload;
275
276 // Firstly we need the first two bytes of the Websocket frame.
277 // This will determinate the size of the frame. This implementation
278 // only supports maximum frame size of 125 characters( bytes, not
279 // unicode characters ) in order to fit in small memory.
280 if( clientBufferCounter < 2 ){
281 clientBuffer[ clientBufferCounter ] = newChar;
282 clientBufferCounter++;
283
284 // Event when the second byte arrives.
285 if( clientBufferCounter == 2 ){
286 // Decode Byte 0
287 decodedHdrFin = ( clientBuffer[ 0 ] >> 7 ) & 0x01;
288 decodedHdrType = (wsDecodedHeaderType_t)( clientBuffer[ 0 ] & 0x0F );
289
290 // Decode Byte 1
291 decodedHdrMask = ( clientBuffer[ 1 ] >> 7 ) & 0x01;
292 decodedHdrPayloadLen = ( clientBuffer[ 1 ] & 0x7F );
293
294 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "---- Data Header Summary ----" ) );
295
296 SHELLMINATOR_WS_DBG( __CONST_TXT__( "\tFin:\t" ) );
297 SHELLMINATOR_WS_DBGLN( decodedHdrFin );
298
299 SHELLMINATOR_WS_DBG( __CONST_TXT__( "\tType:\t" ) );
300 switch( decodedHdrType ){
301 case WS_COMM:
302 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "COMMUNICATION" ) );
303 break;
304 case WS_TXT:
305 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "TEXT" ) );
306 break;
307 case WS_CLOSE:
308 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "CLOSE" ) );
309 break;
310 case WS_PING:
311 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "PING" ) );
312 break;
313 case WS_PONG:
314 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "PONG" ) );
315 break;
316 default:
317 SHELLMINATOR_WS_DBG( __CONST_TXT__( "UNKNOWN CODE: " ) );
318 SHELLMINATOR_WS_DBGLN( (int)decodedHdrType );
319 break;
320 }
321
322 SHELLMINATOR_WS_DBG( __CONST_TXT__( "\tMask:\t" ) );
323 SHELLMINATOR_WS_DBGLN( decodedHdrMask );
324
325 SHELLMINATOR_WS_DBG( __CONST_TXT__( "\tLen:\t" ) );
326 SHELLMINATOR_WS_DBGLN( decodedHdrPayloadLen );
327
328 // The client MUST send mask and mask keys,
329 // if not, we must close the connection.
330 if( !decodedHdrMask ){
331 closeClient();
332 }
333
334 // decodedHdrPayloadCntr is downcounting from decodedHdrPayloadLen
335 decodedHdrPayloadCntr = decodedHdrPayloadLen;
336
337 }
338 }
339
340 // After the first two bytes, the four mask key bytes are coming.
341 else if( decodedHdrMask && ( clientBufferCounter < 6 ) ){
342 clientBuffer[ clientBufferCounter ] = newChar;
343 clientBufferCounter++;
344
345 // Event when the sixth byte arrives.
346 if( clientBufferCounter == 6 ){
347 decodedHdrMaskKeys[ 0 ] = clientBuffer[ 2 ];
348 decodedHdrMaskKeys[ 1 ] = clientBuffer[ 3 ];
349 decodedHdrMaskKeys[ 2 ] = clientBuffer[ 4 ];
350 decodedHdrMaskKeys[ 3 ] = clientBuffer[ 5 ];
351
352 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "\tMask Keys:" ) );
353
354 SHELLMINATOR_WS_DBG( __CONST_TXT__( "\t\t" ) );
355 SHELLMINATOR_WS_DBGLN( decodedHdrMaskKeys[ 0 ], HEX );
356
357 SHELLMINATOR_WS_DBG( __CONST_TXT__( "\t\t" ) );
358 SHELLMINATOR_WS_DBGLN( decodedHdrMaskKeys[ 1 ], HEX );
359
360 SHELLMINATOR_WS_DBG( __CONST_TXT__( "\t\t" ) );
361 SHELLMINATOR_WS_DBGLN( decodedHdrMaskKeys[ 2 ], HEX );
362
363 SHELLMINATOR_WS_DBG( __CONST_TXT__( "\t\t" ) );
364 SHELLMINATOR_WS_DBGLN( decodedHdrMaskKeys[ 3 ], HEX );
365
366 decodedDataXorCntr = 0;
367
368 if( decodedHdrPayloadLen == 0 ){
369 finishDecoding();
370 }
371 else{
372 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "\tDecoded characters:" ) );
373 }
374
375 }
376
377 }
378
379 else if( decodedHdrPayloadCntr > 0 ){
380
381 if( decodedHdrMask ){
382 currentPayload = newChar ^ decodedHdrMaskKeys[ decodedDataXorCntr % 4 ];
383 decodedDataXorCntr++;
384 }
385 else{
386 currentPayload = newChar;
387 }
388
389 SHELLMINATOR_WS_DBG( __CONST_TXT__( "\t\t" ) );
390 SHELLMINATOR_WS_DBG( (char)currentPayload );
391 SHELLMINATOR_WS_DBG( __CONST_TXT__( "\t" ) );
392 SHELLMINATOR_WS_DBG( currentPayload, HEX );
393 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "\tMASKED" ) );
394
395 appendToCircularBuffer( currentPayload );
396
397 decodedHdrPayloadCntr--;
398 if( decodedHdrPayloadCntr == 0 ){
399 finishDecoding();
400 }
401
402 }

406void ShellminatorWebSocket::wsHeaderProcessing( char newChar ){
407 int indexOfSeparator;
408 bool valid;
409
410
411 if( newChar == '\n' ){
412 clientBuffer[ clientBufferCounter ] = '\0';
413 clientBufferCounter = 0;
414 indexOfSeparator = indexOf( clientBuffer, ':' );
415
416 // Detect the end of the header. It consist of an empty line.
417 if( clientBuffer[ 0 ] == '\0' ){
418 // If we found the end of the header, we have to check if all
419 // parameters are present and if they are valid.
420
421 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "WS Client header received." ) );
422
423 valid = true;
424 valid &= connectionUpgradeLineFound;
425 valid &= upgradeWebsocketLineFound;
426 valid &= clientVersion == 13;
427 if( valid ){
428 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "WS Client Header Valid." ) );
429 if( !generateServerKey() ){
430 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "Server key generation failed!" ) );
431 closeClient();
432 return;
433 }
434
435 SHELLMINATOR_WS_DBG( __CONST_TXT__( "WS Server key generated: " ) );
436 SHELLMINATOR_WS_DBGLN( serverKey );
437
438 client.print( serverHeader );
439 client.println( serverKey );
440
441 // We do not send any Sec-WebSocket-Protocol, becouse this
442 // implementation does not support any subprotocols.
443
444 client.println();
445
446 client.flush();
447
448 // Switch state.
449 wsState = WS_CONNECTED_STATE;
450
451 // Reset Buffers for the new state.
452 resetDataVariables();
453
454 SHELLMINATOR_WS_DBG( __CONST_TXT__( "WS Connection Established!" ) );
455
456 if( connectCallback ){
457 connectCallback( this );
458 }
459
460 }
461 else{
462 closeClient();
463 }
464
465 }
466
467 else if( startsWith( clientBuffer, "GET " ) ){
468 SHELLMINATOR_WS_DBG( clientBuffer );
469 SHELLMINATOR_WS_DBG( __CONST_TXT__( "\t\t" ) );
470
471 if( strstr( clientBuffer, "HTTP/1.1" ) != NULL ){
472 httpGetLineFound = true;
473 }
474
475 if( httpGetLineFound ){
476 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "[ OK ]" ) );
477 }
478 else{
479 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "[ NOK ]" ) );
480 }
481 }
482
483 else if( indexOfSeparator > 0 ){
484 // Extract the name of the parameter.
485 strncpy( clientLineNameBuffer, clientBuffer, SHELLMINATOR_WS_CLIENT_LINE_NAME_SIZE );
486
487 // Check if the index of the separator fits in the buffer area.
488 // If not, stop the processing.
489 if( indexOfSeparator > SHELLMINATOR_WS_CLIENT_LINE_NAME_SIZE ){
490 return;
491 }
492 // Terminate the string where the separator was.
493 clientLineNameBuffer[ indexOfSeparator ] = '\0';
494
495 // Extract the value of the parameter.
496 strncpy( clientLineValueBuffer, clientBuffer + indexOfSeparator + 1, SHELLMINATOR_WS_CLIENT_LINE_VALUE_SIZE );
497 // Just in case, terminate the end.
498 clientLineValueBuffer[ SHELLMINATOR_WS_CLIENT_LINE_VALUE_SIZE - 1 ] = '\0';
499
500 // Remove any whitespace from the beginning of the value - RFC2616
501 while( ( clientLineValueBuffer[ 0 ] == ' ' ) || ( clientLineValueBuffer[ 0 ] == '\t' ) ){
502 shiftStringLeft( clientLineValueBuffer );
503 }
504
505 tailEnd( clientLineValueBuffer );
506
507 // Check for 'Connection: Upgrade' line.
508 if( strcicmp( clientLineNameBuffer, "Connection" ) == 0 ){
509 SHELLMINATOR_WS_DBG( clientBuffer );
510 SHELLMINATOR_WS_DBG( __CONST_TXT__( "\t\t" ) );
511
512 if( strcicmp( clientLineValueBuffer, "upgrade" ) == 0 ){
513 connectionUpgradeLineFound = true;
514 }
515
516 if( connectionUpgradeLineFound ){
517 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "[ OK ]" ) );
518 }
519 else{
520 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "[ NOK ]" ) );
521 }
522 }
523
524 // Check for 'Upgrade: websocket' line.
525 else if( strcicmp( clientLineNameBuffer, "Upgrade" ) == 0 ){
526 SHELLMINATOR_WS_DBG( clientBuffer );
527 SHELLMINATOR_WS_DBG( "\t\t" );
528
529 if( strcicmp( clientLineValueBuffer, "websocket" ) == 0 ){
530 upgradeWebsocketLineFound = true;
531 }
532
533 if( upgradeWebsocketLineFound ){
534 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "[ OK ]" ) );
535 }
536 else{
537 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "[ NOK ]" ) );
538 }
539 }
540
541 // Check for 'Sec-WebSocket-Key: XXXXXXXXXXXXXXXXXXXXXXXXX' line.
542 else if( strcicmp( clientLineNameBuffer, "Sec-WebSocket-Key" ) == 0 ){
543 SHELLMINATOR_WS_DBGLN( clientBuffer );
544 strncpy( clientKey, clientLineValueBuffer, SHELLMINATOR_WS_CLIENT_KEY_SIZE );
545 }
546
547 // Check for 'Sec-WebSocket-Version: X' line.
548 else if( strcicmp( clientLineNameBuffer, "Sec-WebSocket-Version" ) == 0 ){
549 SHELLMINATOR_WS_DBGLN( clientBuffer );
550 sscanf( clientLineValueBuffer, "%d", &clientVersion );
551 }
552
553 else{
554 SHELLMINATOR_WS_DBG( __CONST_TXT__( "Undhandled parameter: " ) );
555 SHELLMINATOR_WS_DBGLN( clientBuffer );
556 }
557
558 // There are other fields and I do not know if theye are
559 // necessary in this implementation or not:
560 // - Sec-WebSocket-Extensions
561 // - Sec-WebSocket-Protocol
562 // - Authorization
563
564 }
565 else{
566 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "WS parameter separator not found" ) );
567 }
568
569 }
570 else if( newChar == '\r' ){
571 return;
572 }
573 else{
574 clientBuffer[ clientBufferCounter ] = newChar;
575 clientBufferCounter++;
576 if( clientBufferCounter >= SHELLMINATOR_WS_CLIENT_BUFFER_SIZE ){
577 clientBufferCounter = SHELLMINATOR_WS_CLIENT_BUFFER_SIZE;
578 }
579 }

583bool ShellminatorWebSocket::generateServerKey(){
584 int clientKeySize;
585 int i;
586
587 uint8_t sha1HashBytes[ 20 ];
588
589 clientKeySize = strlen( clientKey );
590 strncpy( clientKey + clientKeySize, "258EAFA5-E914-47DA-95CA-C5AB0DC85B11", SHELLMINATOR_WS_CLIENT_KEY_SIZE );
591 clientKey[ SHELLMINATOR_WS_CLIENT_KEY_SIZE - 1 ] = '\0';
592
593 if( !ShellminatorSHA1( sha1HashBytes, (uint8_t*)clientKey, strlen( clientKey ) ) ){
594 return false;
595 }
596
597 SHELLMINATOR_WS_DBGLN( __CONST_TXT__( "---- SHA1 Hash Bytes ----" ) );
598 for( i = 0; i < sizeof( sha1HashBytes ); i++ ){
599 SHELLMINATOR_WS_DBG( sha1HashBytes[ i ], HEX );
600 }
601 SHELLMINATOR_WS_DBGLN( ' ' );
602
603 if( 0 == ShellminatorBase64( sha1HashBytes, sizeof( sha1HashBytes ), serverKey, SHELLMINATOR_WS_SERVER_KEY_SIZE ) ){
604 return false;
605 }
606
607 return true;

631int ShellminatorWebSocket::available(){
632 if( !clientConnected ){
633 return 0;
634 }
635 if( streamBufferWritePointer == streamBufferReadPointer ){
636 return 0;
637 }
638 else if( streamBufferWritePointer > streamBufferReadPointer ){
639 return streamBufferWritePointer - streamBufferReadPointer;
640 }
641 return SHELLMINATOR_WS_STREAM_BUFFER_SIZE - streamBufferReadPointer + streamBufferWritePointer;
642}
