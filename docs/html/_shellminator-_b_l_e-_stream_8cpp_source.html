71 void ShellminatorBleStream::begin(){
72
73 #ifdef ESP32
74 // Create the BLE Device
75 BLEDevice::init( SHELLMINATOR_BLE_SERVICE_NAME );
76
77 // Create the BLE Server
78 bleServer = BLEDevice::createServer();
79 bleServer -> setCallbacks( &serverCallbacks );
80
81 // Create the BLE Service
82 bleService = bleServer -> createService( NUS_SERVICE_UUID );
83
84 // Create TX characteristics.
85 bleTxChar = bleService -> createCharacteristic( NUS_TX_CHAR_UUID, BLECharacteristic::PROPERTY_NOTIFY );
86 bleTxChar -> addDescriptor( new BLE2902() );
87
88 // Create RX characteristics.
89 bleRxChar = bleService -> createCharacteristic( NUS_RX_CHAR_UUID, BLECharacteristic::PROPERTY_WRITE );
90 bleRxChar -> setCallbacks( &rxCallback );
91
92 // Start the service
93 bleService -> start();
94
95 // Start advertising
96 bleServer -> getAdvertising() -> start();
97
98 SHELLMINATOR_BLE_DBGLN( "NUS service started with ESP32 BLE drivers." );
99
100 #elif defined( ARDUINO_ARCH_NRF52840 ) || defined( ARDUINO_UNOWIFIR4 )
101
102 if( !BLE.begin() ){
103 SHELLMINATOR_BLE_DBGLN( "BLE Init Error!" );
104 return;
105 }
106 BLE.setLocalName( SHELLMINATOR_BLE_SERVICE_NAME );
107 //BLE.setDeviceName( SHELLMINATOR_BLE_SERVICE_NAME );
108
109 BLE.setAdvertisedService( bleService );
110 bleService.addCharacteristic( bleRxChar );
111 bleService.addCharacteristic( bleTxChar );
112 bleRxChar.setEventHandler( BLEWritten, ShellminatorBleStream::bleRxCallback );
113 BLE.addService( bleService );
114 BLE.advertise();
115 BLE.setEventHandler( BLEConnected, onConnect );
116 BLE.setEventHandler( BLEDisconnected, onDisconnect );
117 SHELLMINATOR_BLE_DBGLN( "NUS service started with ESP32 BLE drivers." );
118
119 #endif
120
121 }

123 void ShellminatorBleStream::update(){
124 if( callbackDelayStart != 0 ){
125 if( ( millis() - callbackDelayStart ) > callbackDelayPeriod ){
126 callbackDelayStart = 0;
127 if( state == BLE_CONNECTED_STATE ){
128 if( connectCallback ){
129 connectCallback( this );
130 }
131 }
132 }
133 }
134 #if defined( ARDUINO_ARCH_NRF52840 ) || defined( ARDUINO_UNOWIFIR4 )

165 void ShellminatorBleStream::bleRxCallback::onWrite( BLECharacteristic *charPtr ){
166 int i;
167 int num;
168 std::string dataStr;
169
170 dataStr = charPtr -> getValue();
171 num = dataStr.length();
172
173 if( num > 0 ){
174 SHELLMINATOR_BLE_DBGLN( "Data from BLE client:" );
175 for( i = 0; i < num; i++ ){
176 SHELLMINATOR_BLE_DBG( i );
177 SHELLMINATOR_BLE_DBG( __CONST_TXT__( "\t0x" ) );
178 SHELLMINATOR_BLE_DBG( dataStr[ i ], HEX );
179 SHELLMINATOR_BLE_DBG( __CONST_TXT__( "\t" ) );
180 SHELLMINATOR_BLE_DBGLN( (int)dataStr[ i ] );
181 parent -> appendToCircularBuffer( (uint8_t)dataStr[ i ] );
182 }
183 }

254 void ShellminatorBleStream::appendToCircularBuffer( uint8_t data ){
255 streamBuffer[ streamBufferWritePointer ] = data;
256
257 // Increment the write pointer and handle wrapping.
258 streamBufferWritePointer++;
259 if( streamBufferWritePointer >= SHELLMINATOR_BLE_STREAM_BUFFER_SIZE ){
260 streamBufferWritePointer = 0;
261 }
262
263 // Detect buffer overflow
264 if( streamBufferWritePointer == streamBufferReadPointer ){
265 SHELLMINATOR_BLE_DBGLN( __CONST_TXT__( "BLE Circular Buffer Overflow!" ) );
266 #ifdef ESP32
267 bleService -> stop();
268 #endif
269 }
270 }

279 int ShellminatorBleStream::available(){
280 if( state != BLE_CONNECTED_STATE ){
281 return 0;
282 }
283
284 if( streamBufferWritePointer == streamBufferReadPointer ){
285 return 0;
286 }
287 else if( streamBufferWritePointer > streamBufferReadPointer ){
288 return streamBufferWritePointer - streamBufferReadPointer;
289 }
290 return SHELLMINATOR_BLE_STREAM_BUFFER_SIZE - streamBufferReadPointer + streamBufferWritePointer;
291 }

293 int ShellminatorBleStream::read(){
294 int ret;
295
296 if( state != BLE_CONNECTED_STATE ){
297 return -1;
298 }
299
300 if( streamBufferWritePointer == streamBufferReadPointer ){
301 return -1;
302 }
303 ret = (uint8_t) streamBuffer[ streamBufferReadPointer ];
304 streamBufferReadPointer++;
305
306 if( streamBufferReadPointer >= SHELLMINATOR_BLE_STREAM_BUFFER_SIZE ){
307 streamBufferReadPointer = 0;
308 }
309
310 return ret;
311 }

329 size_t ShellminatorBleStream::write( uint8_t b ){
330 if( state != BLE_CONNECTED_STATE ){
331 return 0;
332 }
333
334 #ifdef ESP32
335 bleTxChar -> setValue( &b, 1 );
336 bleTxChar -> notify();
337 return 1;
338 #elif defined( ARDUINO_ARCH_NRF52840 ) || defined( ARDUINO_UNOWIFIR4 )
339 bleTxChar.writeValue( b );
340 delay( 30 );
341 return 1;
342 #endif
343 }

345 size_t ShellminatorBleStream::write(const uint8_t *data, size_t size){
346 if( state != BLE_CONNECTED_STATE ){
347 return 0;
348 }
349 #ifdef ESP32
350 bleTxChar -> setValue( (uint8_t*) data, size );
351 bleTxChar -> notify();
352 return size;
353 #elif defined( ARDUINO_ARCH_NRF52840 ) || defined( ARDUINO_UNOWIFIR4 )
354 int i = 0;

356 for( i = 0; i < size; i++ ){
357 bleTxBuffer[ cntr ] = data[ i ];
358 cntr++;
359 if( cntr >= SHELLMINATOR_BLE_STREAM_BUFFER_SIZE ){
360 bleTxChar.writeValue( bleTxBuffer, cntr );
361 delay( 30 );
362 cntr = 0;
363 }
364 }
365 bleTxChar.writeValue( bleTxBuffer, cntr );
366 delay( 30 );
367
368 return size;
369 #endif
370 }
