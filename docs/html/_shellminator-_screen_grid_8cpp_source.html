72void ShellminatorScreenGrid::update( int width_p, int height_p ){
73
74 int calculatedOriginX;
75 int calculatedOriginY;
76
77 int calculatedWidth;
78 int calculatedHeight;
79
80 ShellminatorScreen* currentElement;
81
82 Shellminator::shellEvent_t newEvent;
83
84 if( parent == NULL ){
85 return;
86 }
87
88 if( numOfRows < 1 ){
89 return;
90 }
91
92 if( numOfColumns < 1 ){
93 return;
94 }
95
96 width = width_p;
97 height = height_p;
98
99 cellWidth = width / numOfColumns;
100 cellHeight = height / numOfRows;
101
102 newEvent = parent -> readEvent();
103
104 currentElement = head;
105 while( currentElement != NULL ){
106 calculatedOriginX = currentElement -> column * cellWidth + 1;
107 calculatedOriginY = currentElement -> row * cellHeight;
108
109 calculatedWidth = currentElement -> columnSpan * cellWidth;
110 calculatedHeight = currentElement -> rowSpan * cellHeight;
111
112 if( ( calculatedOriginX + calculatedWidth ) > width ){
113 calculatedWidth = width - calculatedOriginX;
114 if( calculatedWidth <= 0 ){
115 currentElement = currentElement -> nextElement;
116 continue;
117 }
118 }
119
120 if( ( calculatedOriginY + calculatedHeight ) > height ){
121 //currentElement = currentElement -> nextElement;
122 //continue;
123 calculatedHeight = height - calculatedOriginY;
124 if( calculatedHeight <= 0 ){
125 currentElement = currentElement -> nextElement;
126 continue;
127 }
128 }
129
130 currentElement -> setOrigin( calculatedOriginX, calculatedOriginY );
131 currentElement -> update( calculatedWidth, calculatedHeight );
132
133 currentElement = currentElement -> nextElement;
134 }
135
136
137 // In case of empty event, we can't do anything.
138 if( newEvent.type == Shellminator::SHELL_EVENT_EMPTY ){
139 return;
140 }
141
142 if( newEvent.type == Shellminator::SHELL_EVENT_RESIZE ){
143 //redraw = true;
144 return;
145 }
146
147
148
149}

151void ShellminatorScreenGrid::draw( bool noClear ){
152
153 int i;
154
155 ShellminatorScreen* currentElement;
156
157 // Check for the necessary parameters.
158 if( parent == NULL ){
159 return;
160 }
161
162 if( channel == NULL){
163 return;
164 }
165
166 if( numOfRows < 1 ){
167 return;
168 }
169
170 if( numOfColumns < 1 ){
171 return;
172 }
173
174 // Only draw if resized event or timer event happened.
175 if( !redraw ){
176 return;
177 }
178 redraw = false;
179
180 for( i = 1; i < height; i++ ){
181 Shellminator::setCursorPosition( channel, 1, i + 1 );
182 channel -> print( "\033[0K" );
183 }
184
185
186 currentElement = head;
187
188 while( currentElement != NULL ){
189 currentElement -> forceRedraw();
190 currentElement -> draw( true );
191 currentElement = currentElement -> nextElement;
192 }
193
194}

229 currentElement = currentElement -> nextElement;
