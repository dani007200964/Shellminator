57void ShellminatorPlot::update( int width_p, int height_p ){
58
59 // Save the widht and height to internal variables.
60 width = width_p;
61 height = height_p;
62
63 Shellminator::shellEvent_t newEvent;
64
65 if( parent == NULL ){
66 return;
67 }
68
69 if( ( millis() - plotTimerStart ) > plotRedrawPeriod ){
70 plotTimerStart = millis();
71 parent -> requestRedraw();
72 redraw = true;
73 }
74
75 newEvent = parent -> readEvent();
76
77 // In case of empty event, we can't do anything.
78 if( newEvent.type == Shellminator::SHELL_EVENT_EMPTY ){
79 return;
80 }
81
82 if( newEvent.type == Shellminator::SHELL_EVENT_RESIZE ){
83 redraw = true;
84 return;
85 }
86
87}

89void ShellminatorPlot::draw( bool noClear ){
90
91 int i;
92 int j;
93
94 int textWidth;
95 int limit;
96
97 if( parent == NULL ){
98 return;
99 }
100
101 if( channel == NULL ){
102 return;
103 }
104
105 // Handle incorrect input data.
106 if( data == NULL ){
107 return;
108 }
109
110 // Handle incorrect input data size.
111 if( dataSize <= 0 ){
112 return;
113 }
114
115 if( name == NULL ){
116 return;
117 }
118
119 // Only draw if resized event or timer event happened.
120 if( !redraw ){
121 return;
122 }
123
124 parent -> clear();
125
126 redraw = false;
127
128 // Calculate the length of the name.
129 textWidth = strlen( name );
130
131 // Set the name text color.
132 parent -> format( channel, Shellminator::BOLD, Shellminator::WHITE );
133
134 // Set cursor to make the text appear on the top center.
135 //Shellminator::setCursorPosition( channel, originX + width / 2 - j / 2, originY );
136 Shellminator::setCursorPosition( channel, originX, originY );
137
138 // Calculate text starting position.
139 // It has to be centered.
140 limit = ( width - textWidth ) / 2;
141
142 // Print as many spaces as needed, to make the text centered.
143 // We print spaces to clear the unwanted garbage from the line.
144 for( i = 0; i < limit; i++ ){
145 channel -> print( ' ' );
146 }
147
148 // Print instructions and plot name only at first draw.
149 channel -> print( name );
150
151 // Calculate how many characters left until line end.
152 limit = width - ( limit + textWidth );
153
154 // Print as many spaces as needed, to make the text centered.
155 // We print spaces to clear the unwanted garbage from the line.
156 for( i = 0; i < limit; i++ ){
157 channel -> print( ' ' );
158 }
159
160 // Set the cursor to the origin position.
161 Shellminator::setCursorPosition( channel, originX, originY );
162
163
164 // Reset variables to a safe state.
165 min = data[ 0 ];
166 max = data[ 0 ];
167
168 // Calculate the min and max values for the scale.
169 for( i = 0; i < dataSize; i++ ){
170
171 if( min > data[ i ] ){
172
173 min = data[ i ];
174
175 }
176
177 if( max < data[ i ] ){
178
179 max = data[ i ];
180
181 }
182
183 }
184
185 // To round down
186 min = ( (int)min ) - 1;
187
188 // To round up
189 max = ( (int)max ) + 1;
190
191 // Draw the scaling grid with numbers.
192 drawScale();
193
194 // Draw the data points.
195 drawPlot();
196
197 // Set the correct style.
198 parent -> format( channel, Shellminator::REGULAR, Shellminator::WHITE );
199
200}

202void ShellminatorPlot::drawScale(){
203
204 int i;
205 int j;
206 int currentTextSize;
207
208 float tmp;
209 float tmpDecimalPart;
210 char sign[2] = { '\0', '\0' };
211
212 // Set the correct style.
213 parent -> format( channel, Shellminator::BOLD, Shellminator::WHITE );
214
215 // Draw the scale.
216 for( i = 0; i < ( height - 1 ); i++ ){
217
218 // Interpolate the value within the height steps.
219 tmp = lerp( max, min, (float)i / (float)( height - 2 ) );
220
221 // Calculate the numbers after the decimal point.
222 // It has to be always positive!
223 tmpDecimalPart = tmp;
224 if( tmpDecimalPart < 0.0 ){
225 tmpDecimalPart *= -1.0;
226 }
227
228 sign[ 0 ] = ' ';
229 if( tmp < 0.0f ){
230 if( (int)tmp == 0 ){
231 sign[ 0 ] = '-';
232 }
233 else{
234 sign[ 0 ] = '\0';
235 }
236 }
237
238 // Generate the string of the number to the data buffer.
239 // Just in case terminate the end of the buffer to avoid
240 // any string problems.
241 snprintf( valueTextBuffer, sizeof( valueTextBuffer ), "%s%d.%d", sign, (int)tmp, (int)( (int)( tmpDecimalPart * 10.0 ) ) % 10 );
242 valueTextBuffer[ sizeof( valueTextBuffer ) - 1 ] = '\0';
243
244 // Calculate the length of the number
245 // string and store it to variable currentTextSize.
246 currentTextSize = strlen( valueTextBuffer );
247
248 // Basic maximum value finding.
249 // If the current text size is greater than the value
250 // in valueTextSizeMax, overwrite valueTextSizeMax with it.
251 // It is required to find the size of the largest printed
252 // number on the Y axes.
253 if( currentTextSize > valueTextSizeMax ){
254 valueTextSizeMax = currentTextSize;
255 redraw = true;
256 }
257
258 // Set the position of the current line.
259 Shellminator::setCursorPosition( channel, originX, originY + 1 + i );
260
261 // Print the actual data string to the output stream.
262 channel -> print( valueTextBuffer );
263 //channel -> print( "\033[0K" );
264
265 }
266
267 // Draw vertical scaling grid
268 Shellminator::setCursorPosition( channel, originX + valueTextSizeMax + 1, originY + 1 );
269
270 for( i = 0; i < ( height - 1 ); i++ ){
271
272 channel -> print( "\u2524" );
273 // Arrow down + left
274 if( i < ( height - 2 ) ){
275 channel -> print( "\033[B\033[D" );
276 }
277
278 }
279
280 // Generate the value string for the last element.
281 // Save the last elements value to the tmp variable.
282 tmp = data[ dataSize - 1 ];
283
284 // Calculate the numbers after the decimal point.
285 // It has to be always positive!
286 tmpDecimalPart = tmp;
287 if( tmpDecimalPart < 0.0 ){
288 tmpDecimalPart *= -1.0;
289 }
290
291 // Generate string from the last element.
292 snprintf( valueTextBuffer, sizeof( valueTextBuffer ), " %d.%d", (int)tmp, (int)( (int)( tmpDecimalPart * 10.0 ) ) % 10 );
293 valueTextBuffer[ sizeof( valueTextBuffer ) - 1 ] = '\0';
294
295 // Calculate the length of the result text.
296 resultTextSize = strlen( valueTextBuffer );
297
298 // Set the cursor to the result scaling grid position.
299 Shellminator::setCursorPosition( channel, originX + width - resultTextSize - 2, originY + 1 );
300
301 // Draw vertical scaling grid to the end for the result.
302 for( i = 0; i < ( height - 1 ); i++ ){
303
304 channel -> print( "\u251C" );
305
306 // Arrow down + left
307 if( i < ( height - 2 ) ){
308 channel -> print( "\033[B\033[D" );
309 }
310
311 }

316 return( 1.0 - t ) * v0 + t * v1;

319float ShellminatorPlot::mapFloat( float x, float inStart, float inStop, float outStart, float outStop ){
320
321 return outStart + ( outStop - outStart ) * ( ( x - inStart ) / ( inStop - inStart ) );

326void ShellminatorPlot::drawPlot(){

345

350 /*

353

357
358 // Find the actual data point for the current horizontal position.
359 horizontalIndex = lerp( 0, dataSize, (float)j / terminalWidth );
360 verticalIndex = mapFloat( data[ horizontalIndex ], min, max, height-1, 0 );
361 //Shellminator::setCursorPosition( channel, originX + valueTextSizeMax + 3 + i - 1, originY + verticalIndex );
362 //channel -> print( '*' );
363
364 nextChar = clearCell;
365
366 if( j < ( terminalWidth - 1 ) ){
367 horizontalIndexNext = lerp( 0, dataSize, (float)( j + 1 ) / terminalWidth );
368 verticalIndexNext = mapFloat( data[ horizontalIndexNext ], min, max, height-1, 0 );
369 }
370
371 if( verticalIndex == i ){
372 //channel -> print( "\u2022" );
373 nextChar = dotCell;
374 }
375
376 else if( ( verticalIndex < i ) && ( verticalIndexNext > i ) ){
377 //channel -> print( '+' );
378 nextChar = dotCell;
379 }
380
381 else if( ( verticalIndex > i ) && ( verticalIndexNext < i ) ){
382 //channel -> print( '+' );
383 nextChar = dotCell;
384 }
385
386 channel -> print( nextChar );
387
388
389 }
390
391 // Cursor right with one characters.
392 channel -> print( "\033[C" );
393
394 // Check if it is the most right data point.
395 if( verticalIndex == i ){
396 // If it is, print the result value.
397 channel -> print( valueTextBuffer );
398 }
399 else{
400 // If it is not, print as many whitespace
401 // characters, as long the result text.
402 for( j = 0; j < resultTextSize; j++ ){
403 channel -> print( ' ' );
404 }
405 }
406
407 }
