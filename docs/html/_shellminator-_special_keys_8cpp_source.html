124void Shellminator::ShellminatorUpArrowKeyState(){
125
126 // Because we have finished the escape sequence interpretation we reset the state-machine.
127 currentState = &Shellminator::ShellminatorDefaultState;
128
129 if( !loggedIn ){
130 return;
131 }
132 if( inputActive ){
133 return;
134 }
135
136 if( screen ){
137 pushEvent( ( shellEvent_t ){ SHELL_EVENT_CODED_KEY, EVENT_CODE_UP_ARROW } );
138 return;
139 }
140
141 // Check if the arrow function is overriden.
142 if( upArrowOverrideFunc ){
143 upArrowOverrideFunc( this );
144 return;
145 }
146
147 // We have to check that we can go upper in history
148 if ( cmd_buff_dim < ( SHELLMINATOR_BUFF_DIM ) ) {
149
150 // If we can, we have to check that the previous command was not empty.
151 if ( cmd_buff[ cmd_buff_dim ][0] == '\0' ) {
152 // If it was empty we can't do much with an empty command so we return.
153 return;
154 }
155
156 // Now we have to copy the characters form the histoy to the 0th element in the buffer.
157 // Remember the 0th element is always reserved for the new data. If we browse the history the
158 // data in the history will overwrite the data in the 0th element so the historical data will be
159 // the new data. We use strncpy to prevent overflow.
160 strncpy( cmd_buff[ 0 ], cmd_buff[ cmd_buff_dim ], SHELLMINATOR_BUFF_LEN + 1 );
161
162 // We have to calculate the historical data length to pass it to the cmd_buff_cntr variable.
163 // It is important to track the end of the loaded string.
164 cmd_buff_cntr = strlen( cmd_buff[ 0 ] );
165 cursor = cmd_buff_cntr;
166
167 // We print the loaded command to the terminal interface.
168 //channel -> print( cmd_buff[ 0 ] );
169
170 redrawLine();
171
172 // We have to increment the cmd_buff_dim variable, to track the history position.
173 // Greater number means older command!
174 cmd_buff_dim++;
175
176 }
177
178}

180void Shellminator::ShellminatorDownArrowKeyState(){
181
182 // Because we have finished the escape sequence interpretation we reset the state-machine.
183 currentState = &Shellminator::ShellminatorDefaultState;
184
185 if( !loggedIn ){
186 return;
187 }
188 if( inputActive ){
189 return;
190 }
191
192 if( screen ){
193 pushEvent( ( shellEvent_t ){ SHELL_EVENT_CODED_KEY, EVENT_CODE_DOWN_ARROW } );
194 return;
195 }
196
197 // Check if the arrow function is overriden.
198 if( downArrowOverrideFunc ){
199 downArrowOverrideFunc( this );
200 return;
201 }
202
203 // We have to check that we can go lover in history, and we are not in the first previous command.
204 if ( cmd_buff_dim > 2 ) {
205
206 // We have to decrement the cmd_buff_dim variable, to track the history position.
207 // Lower number means newer command!
208 cmd_buff_dim--;
209
210 // Now we have to copy the characters form the histoy to the 0th element in the buffer.
211 // Remember the 0th element is always reserved for the new data. If we browse the history the
212 // data in the history will overwrite the data in the 0th element so the historical data will be
213 // the new data. We use strncpy to prevent overflow.
214 strncpy( cmd_buff[ 0 ], cmd_buff[ cmd_buff_dim - 1 ], SHELLMINATOR_BUFF_LEN + 1 );
215
216 // We have to calculate the historical data length to pass it to the cmd_buff_cntr variable.
217 // It is important to track the end of the loaded string.
218 cmd_buff_cntr = strlen( cmd_buff[ 0 ] );
219 cursor = cmd_buff_cntr;
220
221 // We print the loaded command to the terminal interface.
222 //channel -> print( cmd_buff[ 0 ] );
223 redrawLine();
224
225 }
226
227 // Check that if we are in the first previous command.
228 else if ( cmd_buff_dim == 2 ) {
229
230 // To empty the incoming string we have to zero it's counter.
231 cmd_buff_cntr = 0;
232 cursor = 0;
233
234 // We have to reset the cmd_buff_dim variable to the default value.
235 cmd_buff_dim = 1;
236
237 redrawLine();
238
239 }
240
241 // We have finished so we can break from the switch.
242
243}

245void Shellminator::ShellminatorLeftArrowKeyState(){
246
247 // We just simply reset the state-machine.
248 currentState = &Shellminator::ShellminatorDefaultState;
249
250 if( screen ){
251 pushEvent( ( shellEvent_t ){ SHELL_EVENT_CODED_KEY, EVENT_CODE_LEFT_ARROW } );
252 return;
253 }
254
255 // Check if the arrow function is overriden.
256 if( leftArrowOverrideFunc && loggedIn ){
257 leftArrowOverrideFunc( this );
258 return;
259 }
260
261 // Detect if the channel was configured incorrectly.
262 // In this case, we have to stop command execution.
263 if( channel == NULL ){
264 return;
265 }
266
267 // Check if we can move to left.
268 if( cursor > 0 ){
269
270 channel -> write( 27 ); // ESC character( decimal 27 )
271 channel -> print( '[' ); // VT100 Cursor command.
272 channel -> print( '1' ); // 1 character movement.
273 channel -> print( 'D' ); // Left.
274
275 // Decrement the cursor variable.
276 cursor--;
277
278 }

391void Shellminator::ShellminatorDelKeyState(){
392
393 // Del key detected.
394 // If we press a delet key we have to reset cmd_buff_dim to default value
395 cmd_buff_dim = 1;
396
397 // General counter variable
398 uint32_t i;
399
400 currentState = &Shellminator::ShellminatorDefaultState;
401
402 if( screen != NULL ){
403 return;
404 }
405
406 // We have to check the number of the characters in the buffer.
407 // If the buffer is full we must not do anything!
408 if ( cursor != cmd_buff_cntr ) {
409
410 for( i = cursor; i < ( cmd_buff_cntr - 1 ); i++ ){
411 cmd_buff[ 0 ][ i ] = cmd_buff[ 0 ][ i + 1 ];
412 }
413
414 // If there is at least 1 character in the buffer we jus simply
415 // decrement the cmd_buff_cntr. This will result that the new character
416 // will be stored in the previous characters place in the buffer.
417 cmd_buff_cntr--;
418
419 redrawLine();
420
421 }
