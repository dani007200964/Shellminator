69 void ShellminatorTcpSocket::appendToCircularBuffer( uint8_t data ){
70 streamBuffer[ streamBufferWritePointer ] = data;
71
72 // Increment the write pointer and handle wrapping.
73 streamBufferWritePointer++;
74 if( streamBufferWritePointer >= SHELLMINATOR_TCP_STREAM_BUFFER_SIZE ){
75 streamBufferWritePointer = 0;
76 }
77
78 // Detect buffer overflow
79 if( streamBufferWritePointer == streamBufferReadPointer ){
80 SHELLMINATOR_TCP_DBGLN( __CONST_TXT__( "TCP Circular Buffer Overflow!" ) );
81 closeClient();
82 }

93 void ShellminatorTcpSocket::update(){
94 char newChar;
95 WiFiClient newClient;
96
97 if( server == NULL ){
98 return;
99 }
100
101 newClient = server -> accept();
102
103 if( newClient ){
104 if( CLIENT_STATE ){
105 // Reject a new client, because another one is alredy in use.
106 newClient.stop();
107 SHELLMINATOR_TCP_DBGLN( __CONST_TXT__( "TCP Client already in use. Rejecting new connection." ) );
108 }
109 else{
110 // New connection.
111 client = newClient;
112 #ifndef ARDUINO_UNOWIFIR4
113 client.setNoDelay( true );
114 #endif
115 clientConnected = true;
116 SHELLMINATOR_TCP_DBGLN( __CONST_TXT__( "New TCP Client." ) );
117 SHELLMINATOR_TCP_DBGLN( __CONST_TXT__( "---- Sending Telnet Configuration ----" ) );
118
119 client.write( TELNET_IAC_DONT_LINEMODE, 3 );
120 client.write( TELNET_IAC_WILL_ECHO, 3 );
121 client.write( TELNET_IAC_DONT_ECHO, 3 );
122 client.write( TELNET_IAC_WILL_SUPPRESS_GO_AHEAD, 3 );
123 client.write( TELNET_IAC_DO_SUPPRESS_GO_AHEAD, 3 );
124
125 if( connectCallback ){
126 connectCallback( this );
127 }
128 }
129 }
130
131 // Check for disconnection event
132 if( clientConnected && !CLIENT_STATE ){
133 SHELLMINATOR_TCP_DBGLN( __CONST_TXT__( "TCP Disconnect Event Detected!" ) );
134 closeClient();
135 }
136
137 // Process webscoket data.
138 if( clientConnected && CLIENT_STATE ){
139
140 while( client.available() ){
141 newChar = client.read();
142
143 switch( negotiationState ){
144 case TELNET_FIRST_BYTE_STATE:
145 // Check if we found an ongoing Telnet command
146 if( newChar == 0xFF ){
147 // If so, we have to change state. Each Telnet
148 // command starts with 0xFF and after this,
149 // two bytes follow.
150 negotiationState = TELNET_COMMAND_BYTE_STATE;
151 }
152 else{
153 appendToCircularBuffer( newChar );
154 }
155 break;
156 case TELNET_COMMAND_BYTE_STATE:
157 // The current implementation does not care about the
158 // Telnet commands. But to not throw any garbage characters
159 // in front of the user, we read the command and option
160 // bytes to remove them from the buffer.
161 negotiationState = TELNET_OPTION_BYTE_STATE;
162 break;
163 case TELNET_OPTION_BYTE_STATE:
164 // See the comment above.
165 negotiationState = TELNET_FIRST_BYTE_STATE;
166 break;
167 default:
168 // We should not be here!
169 closeClient();
170 break;
171 }
172
173 }
174
175 }

179 void ShellminatorTcpSocket::closeClient(){
180 client.flush();
181 delay( 100 );
182 while( client.available() ){
183 client.read();
184 }
185 client.stop();
186 clientConnected = false;
187 resetVariables();
188 SHELLMINATOR_TCP_DBGLN( __CONST_TXT__( "Disconnecting TCP Client!" ) );
189 if( disconnectCallback ){
190 disconnectCallback( this );

203 int ShellminatorTcpSocket::available(){
204 if( !clientConnected ){
205 return 0;
206 }
207 if( streamBufferWritePointer == streamBufferReadPointer ){
208 return 0;
209 }
210 else if( streamBufferWritePointer > streamBufferReadPointer ){
211 return streamBufferWritePointer - streamBufferReadPointer;
212 }
213 return SHELLMINATOR_TCP_STREAM_BUFFER_SIZE - streamBufferReadPointer + streamBufferWritePointer;
214 }

216 int ShellminatorTcpSocket::read(){
217 int ret;
218 if( !clientConnected ){
219 return -1;
220 }
221 if( streamBufferWritePointer == streamBufferReadPointer ){
222 return -1;
223 }
224 ret = (uint8_t) streamBuffer[ streamBufferReadPointer ];
225 streamBufferReadPointer++;
226
227 if( streamBufferReadPointer >= SHELLMINATOR_TCP_STREAM_BUFFER_SIZE ){
228 streamBufferReadPointer = 0;
229 }
230
231 return ret;
232 }
