38void ShellminatorQR::generate( Stream* channel_p, const char* text, enum qrcodegen_Ecc ecc ){
39
40 // The result of the QR-code generation will be stored in this variable.
41 bool result;
42
43 // The actual size of the QR-code will be stored in this variable.
44 uint32_t qr_size;
45
46 // This variable helps to track the x direction while drawing the QR code.
47 uint32_t x;
48
49 // This variable helps to track the y direction while drawing the QR code.
50 uint32_t y;
51
52 // This variable will store the upper pixel while drawing.
53 bool upper_pixel;
54
55 // This variable will store the lower pixel while drawing.
56 bool lower_pixel;
57
58 // Detect if the channel was configured incorrectly.
59 // In this case, we have to stop command execution.
60 if( channel_p == NULL ){
61 return;
62 }
63
64 // Generate the QR-code with default settings, and store the result to
65 // the result variable.
66 result = qrcodegen_encodeText( text,
67 qr_tempBuff,
68 qr_data,
69 ecc,
70 qrcodegen_VERSION_MIN,
71 qrcodegen_VERSION_MAX,
72 qrcodegen_Mask_AUTO,
73 true
74 );
75
76 // We have to check the result variable. If it is true,
77 // the QR-code generation went well.
78 if( !result ){
79 return;
80 }
81
82 // Determinate the size of the QR-code.
83 qr_size = qrcodegen_getSize( qr_data );
84
85 // The unicode character table does not have a rectangular square,
86 // but it has a half rectangular square on, top, bottom, and a full bar.
87 // Check these links to make it more clear:
88 // -Full bar: https://www.fileformat.info/info/unicode/char/2588/index.htm
89 // -Upper square: https://www.fileformat.info/info/unicode/char/2580/index.htm
90 // -lower square: https://www.fileformat.info/info/unicode/char/2584/index.htm
91 // To draw a QR-code with a terminal emulator the easiest way to combine these
92 // unicode characters. Because it is two 'pixels' high, we have to step the y
93 // variable by two lines.
94 for( y = 0; y < ( qr_size / 2 ); y++ ){
95
96 // Print a new line at the begining of the horizontal drawing.
97 channel_p -> print( "\r\n" );
98
99 // Draw all horizontal 'pixels'.
100 for( x = 0; x < qr_size; x++ ){
101
102 // Determinate the upper pixel value.
103 upper_pixel = qrcodegen_getModule( qr_data, x, ( y * 2 ) );
104
105 // Determinate the lower pixel value.
106 lower_pixel = qrcodegen_getModule( qr_data, x, ( ( y * 2 ) + 1 ) );
107
108 // Draw the right pattern accordingly.
109
110 // Both pixels are black.
111 if( upper_pixel && lower_pixel ){
112
113 channel_p -> print( "\u2588" );
114 continue;
115
116 }
117
118 // Upper pixel is black.
119 if( upper_pixel && ( !lower_pixel ) ){
120
121 channel_p -> print( "\u2580" );
122 continue;
123
124 }
125
126 // Lower pixel is black.
127 if( ( !upper_pixel ) && lower_pixel ){
128
129 channel_p -> print( "\u2584" );
130 continue;
131
132 }
133
134 // If we get here we have to draw an empty bar.
135 // The space character will do the job.
136 channel_p -> print( " " );
137
138 }
139
140 }
141
142 // If the QR code size is not even, we have to draw
143 // the last line as well.
144 if( ( qr_size % 2 ) != 0 ){
145
146 // Print a new line at the begining of the horizontal drawing.
147 channel_p -> print( "\r\n" );
148
149 // Draw all horizontal 'pixels'.
150 for( x = 0; x < qr_size; x++ ){
151
152 // Determinate the pixel value. We store it to upper_pixel variable.
153 upper_pixel = qrcodegen_getModule( qr_data, x, ( qr_size - 1 ) );
154
155 // Check if we have to draw.
156 if( upper_pixel ){
157
158 channel_p -> print( "\u2580" );
159 continue;
160
161 }
162
163 // If we get here we have to draw an empty bar.
164 // The space character will do the job.
165 channel_p -> print( " " );
166
167 }
168
169 }
170
171 // Finally create a new line.
172 channel_p -> print( "\r\n" );
173
174}
