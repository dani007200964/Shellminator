73void ShellminatorButton::attachEvent( Shellminator::shellEvent_t event_p ){
74
75 // Save the trigger event struct.
76 event = event_p;
77
78 // If trigger event is a regular key, save the data to internal variables.
79 if( event_p.type == Shellminator::SHELL_EVENT_KEY ){
80 if( isUpper( (char)event.data ) ){
81 eventText[ 0 ] = 0xE2;
82 eventText[ 1 ] = 0x86;
83 eventText[ 2 ] = 0x91;
84 eventText[ 3 ] = ' ';
85 eventText[ 4 ] = (char)event.data;
86 eventText[ 5 ] = '\0';
87 eventTextSize = 3;
88 }
89
90 else if( isLower( (char)event.data ) ){
91 eventText[ 0 ] = (char)event.data;
92 eventText[ 1 ] = '\0';
93 eventTextSize = 1;
94 }
95
96 else{
97 event.type = Shellminator::SHELL_EVENT_EMPTY;
98 eventTextSize = 0;
99 return;
100 }
101
102 }
103
104 // Otherwise, we can not enable trigger event functionality.
105 else{
106 event.type = Shellminator::SHELL_EVENT_EMPTY;
107 eventTextSize = 0;
108 return;
109 }
110
111}

126void ShellminatorButton::update( int width_p, int height_p ){
127
128 Shellminator::shellEvent_t newEvent;
129
130 if( parent == NULL ){
131 return;
132 }
133
134 width = width_p;
135 height = height_p;
136
137 // If the user specified width is too small,
138 // overwrite it with valid configuration.
139 if( width < ( textWidth + 4 ) ){
140 width = textWidth + 4;
141 }
142
143 // The height must be exactly 3.
144 if( height != 3 ){
145 height = 3;
146 }
147
148 newEvent = parent -> readEvent();
149
150 // In case of empty event, we can't do anything.
151 if( newEvent.type == Shellminator::SHELL_EVENT_EMPTY ){
152 return;
153 }
154
155 if( newEvent.type == Shellminator::SHELL_EVENT_RESIZE ){
156 redraw = true;
157 return;
158 }
159
160 // If the event is key related, we have to check if the key
161 // matches with the attached key.
162 if( ( newEvent.type == Shellminator::SHELL_EVENT_KEY ) && ( newEvent.type == event.type ) ){
163 if( newEvent.data == event.data ){
164 if( func != NULL ){
165 func( this );
166 }
167 return;
168 }
169 }
170
171 // Check if the event was a left mouse release.
172 if( ( newEvent.type == Shellminator::SHELL_EVENT_MOUSE ) && ( newEvent.eventCode == Shellminator::EVENT_CODE_MOUSE_LEFT_RELEASED ) ){
173
174 // Check if the event happened in the area of the button.
175 if( ( newEvent.x >= originX ) &&
176 ( newEvent.y >= originY ) &&
177 ( newEvent.x < ( originX + width ) ) &&
178 ( newEvent.y < ( originY + height ) ) ){
179
180 if( func != NULL ){
181 func( this );
182 }
183 return;
184 }
185
186 }

190void ShellminatorButton::draw(){
191
192 // Generic counter.
193 int i;
194
195 int printedEventTextSize = 0;
196
197 // The upper limit of the for loops calculated
198 // to this variable.
199 int limit;
200
201 if( parent == NULL ){
202 return;
203 }
204
205 if( channel == NULL){
206 return;
207 }
208
209 if( name == NULL ){
210 return;
211 }
212
213 // Only draw if resized event or timer event happened.
214 if( !redraw ){
215 return;
216 }
217 redraw = false;
218
219 // Top line section.
220 Shellminator::setCursorPosition( channel, originX, originY );
221
222 Shellminator::setFormat( channel, Shellminator::REGULAR, color );
223
224 channel -> print( __CONST_TXT__( "\u250C" ) );
225
226 for( i = 2; i < width; i++ ){
227 channel -> print( __CONST_TXT__( "\u2500" ) );
228 }
229
230 channel -> print( __CONST_TXT__( "\u2510" ) );
231
232 // Middle line section.
233 Shellminator::setCursorPosition( channel, originX, originY + 1 );
234
235 channel -> print( __CONST_TXT__( "\u2502" ) );
236
237 if( eventTextSize > 0 ){
238 printedEventTextSize = eventTextSize + 6;
239 }
240 limit = ( width - ( textWidth + 3 + printedEventTextSize ) ) / 2;
241 for( i = 1; i < limit; i++ ){
242 channel -> print( ' ' );
243 }
244
245 Shellminator::setFormat( channel, Shellminator::BOLD, Shellminator::WHITE );
246
247 channel -> print( __CONST_TXT__( "\u2BA9 " ) );
248 channel -> print( name );
249
250 if( eventTextSize > 0 ){
251 channel -> print( __CONST_TXT__( " [ " ) );
252 channel -> print( eventText );
253 channel -> print( __CONST_TXT__( " ]" ) );
254 }
255
256 Shellminator::setFormat( channel, Shellminator::REGULAR, color );
257
258 limit = width - ( limit + textWidth + 3 + printedEventTextSize );
259 for( i = 1; i < limit; i++ ){
260 channel -> print( ' ' );
261 }
262
263 channel -> print( __CONST_TXT__( "\u2502" ) );
264
265 // Bottom line section.
266 Shellminator::setCursorPosition( channel, originX, originY + 2 );
267
268 channel -> print( __CONST_TXT__( "\u2514" ) );
269
270 for( i = 2; i < width; i++ ){
271 channel -> print( __CONST_TXT__( "\u2500" ) );
272 }
273
274 channel -> print( __CONST_TXT__( "\u2518" ) );
275
276}

278bool ShellminatorButton::isUpper( char c ){
279 return ( c >= 'A' ) && ( c <= 'Z' );
280}

282bool ShellminatorButton::isLower( char c ){
283 return ( c >= 'a' ) && ( c <= 'z' );

286char ShellminatorButton::toUpper( char c ){
287 if( isLower( c ) ){
288 // Check ASCII table to decode this magic.
289 return c - ( 'a' - 'A' );
290 }
291 return c;
