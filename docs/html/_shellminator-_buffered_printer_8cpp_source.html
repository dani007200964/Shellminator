52bool ShellminatorBufferedPrinter::allocate( int bufferSize_p ){
53
54 // We need at least 30 characters to work properly.
55 if( bufferSize_p < 30 ){
56 bufferSize = -1;
57 return false;
58 }
59
60 bufferSize = bufferSize_p;
61
62 availableCharacters = bufferSize;
63 acceleratorBuffer = (char*)malloc( bufferSize * sizeof( char ) );
64 acceleratorBufferPointer = (char*)acceleratorBuffer;
65
66 /*
67 char buff[ 200 ];
68 snprintf( buff, sizeof( buff ), "allocated address: %p\r\n", acceleratorBuffer );
69 channel -> print( buff );
70 */
71
72 if( acceleratorBuffer == NULL ){
73 // Memory allocation failed!
74 bufferSize = -1;
75 return false;
76 }
77
78 clearBuffer();
79 return true;
80
81}

121int ShellminatorBufferedPrinter::printf( const char *fmt, ... ){
122
123 va_list args;
124
125 int status;
126
127 char* pointerBeforePrint;
128
129 // Save the pointer position before any printing.
130 // If the buffer gets full, we have to protect the
131 // data already in the buffer.
132 pointerBeforePrint = acceleratorBufferPointer;
133
134 // Start the variadic list.
135 va_start( args, fmt );
136
137 // If memory allocation failed with the constructor.
138 if( ( bufferSize < 0 ) || ( channel == NULL ) ){
139
140 // Close the variadic list and return.
141 va_end( args );
142 return 0;
143
144 }
145
146 // Try to print the whole data into the buffer.
147 status = vsnprintf( acceleratorBufferPointer, availableCharacters, fmt, args );
148 //wprintf( L"Available characters: %d\r\n", availableCharacters );
149
150 // Check if we ran out of memory.
151 if( status >= availableCharacters ){
152
153 // We possibly ran out of buffer.
154 // Flush the buffer and try again.
155 flush();
156
157 // Try again with empty buffer.
158 status = vsnprintf( acceleratorBufferPointer, availableCharacters, fmt, args );
159
160 // Check the memory again.
161 if( status >= availableCharacters ){
162
163 // If we got a problem with empty buffer, we can not print this data,
164 // because the buffer is too small.
165
166 // Reset the pointer and the counter for the next transaction.
167 acceleratorBufferPointer = acceleratorBuffer;
168 availableCharacters = bufferSize;
169
170 va_end( args );
171 return 0;
172
173 }
174
175 }
176
177 acceleratorBufferPointer += status;
178 availableCharacters -= status;
179
180
181 va_end( args );
182
183 return status;
184
185}

187void ShellminatorBufferedPrinter::flush(){
188
189 // If memory allocation failed with the constructor.
190 if( ( bufferSize < 0 ) || ( channel == NULL ) ){
191
192 return;
193
194 }
195
196 // Terminate the string.
197 *acceleratorBufferPointer = '\0';
198
199 // Print the data.
200 channel -> print( acceleratorBuffer );
201
202 // Reset the pointer and the counter.
203 acceleratorBufferPointer = acceleratorBuffer;
204 availableCharacters = bufferSize;
205
206 clearBuffer();
207
208}

210void ShellminatorBufferedPrinter::clearBuffer(){
211
212 int i;
213
214 // If memory allocation failed with the constructor.
215 if( ( bufferSize < 0 ) || ( channel == NULL ) ){
216
217 return;
218
219 }
220
221 for( i = 0; i < bufferSize; i++ ){
222
223 acceleratorBuffer[ i ] = '\0';
224
225 }

227}
