46void ShellminatorProgress::draw( bool noClear ){
47
48 // Ratio used to draw[ 0.0 -1.0 ].
49 float ratio;
50
51 // Generic counter
52 int i;
53
54 // Used to calculate the target for the for loops.
55 int limit;
56
57 // The printed text size will be calculated to this variable.
58 int textSize;
59
60 // Used for remaining time calculation.
61 int remainingTime;
62
63 if( parent == NULL ){
64 return;
65 }
66
67 if( channel == NULL){
68 return;
69 }
70
71 // Only draw if resized event or timer event happened.
72 if( !redraw ){
73 return;
74 }
75 redraw = false;
76
77 // Set cursor to top left.
78 Shellminator::setCursorPosition( channel, originX, originY );
79
80 // Copy percentage value to ratio.
81 ratio = percentage;
82
83 // Limit the ratio to a valid range.
84 if( ratio > 100.0f ){
85 ratio = 100.0f;
86 }
87
88 if( ratio < 0.0f ){
89 ratio = 0.0f;
90 }
91
92 // For padding.
93 if( ( (int)ratio ) < 100 ){
94 channel -> print( ' ' );
95 }
96
97 if( ( (int)ratio ) < 10 ){
98 channel -> print( ' ' );
99 }
100
101 // Print percentage.
102 channel -> print( (int)ratio );
103 channel -> print( '.' );
104 channel -> print( (int)((float)ratio * 10.0) % 10 );
105 channel -> print( "%\u2502" );
106
107 // Transform ratio to 0.0 - 1.0.
108 ratio = ratio / 100.0;
109
110 // If we do not have a text or a format, we just
111 // render the progress bar.
112 if( ( text == NULL ) && ( format == NULL ) ){
113
114 // Calculate how many 'bars' has to be rendered.
115 limit = (float)( width - 9 ) * ratio + 9;
116
117 // Set the color to the specified one.
118 parent -> format( channel, color );
119
120 // It has to start from 9 to avoid xTerm glitch.
121 for( i = 9; i < limit; i++ ){
122
123 // Decide to render a bar or a blank space.
124 if( i < limit ){
125 channel -> print( "\u2588" );
126 }
127
128 else{
129 channel -> print( ' ' );
130 }
131
132 }
133
134 // Set back formatting to regular.
135 parent -> format( channel, Shellminator::WHITE );
136
137 // Draw progress bar end character.
138 channel -> print( "\u2502" );
139
140 }
141
142 else{
143
144 // Calculate how many 'bars' has to be rendered.
145 limit = (float)( width / 2 - 10 ) * ratio + 10;
146
147 // Set the color to the specified one.
148 parent -> format( channel, color );
149
150 // It has to start from 10 to avoid xTerm glitch.
151 for( i = 10; i < ( width / 2 ); i++ ){
152
153 // Decide to render a bar or a blank space.
154 if( i < limit ){
155 channel -> print( "\u2588" );
156 }
157
158 else{
159 channel -> print( ' ' );
160 }
161
162 }
163
164 // Set back formatting to regular.
165 parent -> format( channel, Shellminator::WHITE );
166
167 // Draw progress bar end character.
168 channel -> print( "\u2502 " );
169
170 // Reset text size.
171 textSize = 0;
172
173 // Check if a format is specified or not.
174 if( format != NULL ){
175
176 // Go through every format character and print according.
177 for( i = 0; i < strlen( format ); i++ ){
178
179 switch( format[ i ] ){
180
181 // Regular separator.
182 case '|':
183 textSize += channel -> print( '|' );
184 break;
185
186 // Blank space.
187 case ' ':
188 textSize += channel -> print( ' ' );
189 break;
190
191 // Seconds since start event.
192 case 's':
193 textSize += channel -> print( ( millis() - timerStart ) / 1000 );
194 textSize += channel -> print( "s " );
195 break;
196
197 // Minutes since start event.
198 case 'm':
199 textSize += channel -> print( ( millis() - timerStart ) / 1000 / 60 );
200 textSize += channel -> print( "m " );
201 break;
202
203 // Time since start
204 case 't':
205 // I know, the naming is not the best, but I need a buffer variable.
206 remainingTime = ( ( millis() - timerStart ) / 1000 ); // <- seconds since start event.
207 textSize += channel -> print( ( millis() - timerStart ) / 1000 / 60 );
208
209 // Print remaining minutes.
210 if( ( remainingTime / 60 ) < 10 ){
211 // For padding.
212 textSize += channel -> print( '0' );
213 }
214 textSize += channel -> print( remainingTime / 60 );
215 textSize += channel -> print( ':' );
216 remainingTime %= 60;
217
218 // Print remaining seconds
219 if( ( remainingTime ) < 10 ){
220 // For padding.
221 textSize += channel -> print( '0' );
222 }
223 textSize += channel -> print( remainingTime );
224 textSize += channel -> print( ' ' );
225
226 break;
227
228 // Remaining time.
229 case 'r':
230 // We have to check if the ratio is within boundaries
231 // to avoid division by zero.
232 if( ( ratio > 0.0 ) && ( ratio < 1.0 ) ){
233
234 // Calculate remaining seconds.
235 remainingTime = ( ( millis() - timerStart ) / 1000 );
236 remainingTime = remainingTime * ( 1.0 / ratio ) - remainingTime;
237
238 // Print remaining minutes.
239 if( ( remainingTime / 60 ) < 10 ){
240 // For padding.
241 textSize += channel -> print( '0' );
242 }
243 textSize += channel -> print( remainingTime / 60 );
244 textSize += channel -> print( ':' );
245 remainingTime %= 60;
246
247 // Print remaining seconds
248 if( ( remainingTime ) < 10 ){
249 // For padding.
250 textSize += channel -> print( '0' );
251 }
252 textSize += channel -> print( remainingTime );
253
254 }
255 else{
256 // If we are out of range print invalid timing data.
257 textSize += channel -> print( "-:-" );
258 }
259
260 // Finish the message.
261 textSize += channel -> print( " left " );
262 break;
263
264 default:
265 break;
266 }
267
268 }
269
270 }
271
272 // Check if we have a message to print.
273 if( text != NULL ){
274
275 // Calculate message size and add it to the
276 // already existing size.
277 textSize += strlen( text );
278
279 // Calculate how many characters left in the row.
280 limit = width - ( width / 2 );
281
282 // Check if the limit is greater, than the text.
283 if( limit > textSize ){
284 // We can not print more characters than the
285 // text contains.
286 limit = textSize;
287 }
288
289 // Print as many characters from the text as possible.
290 for( i = 0; i < limit; i++ ){
291 // We have to check string termination.
292 if( text[ i ] == '\0' ){
293 // If we reached string end, we clear the rest of
294 // of the line and return.
295 if( !noClear ){
296 channel -> print( "\033[0K" );
297 }
298 return;
299 }
300
301 // Print next character from the string.
302 channel -> print( text[ i ] );
303 }
304
305 }
306
307 }
308
309 // Clear the rest of the line.
310 if( !noClear ){
311 channel -> print( "\033[0K" );
312 }
313
314}

316void ShellminatorProgress::update( int width_p, int height_p ){
317
318 Shellminator::shellEvent_t newEvent;
319
320 if( parent == NULL ){
321 return;
322 }
323
324 width = width_p;
325 height = height_p;
326
327 if( height > 1 ){
328 height = 1;
329 }
330
331 // Try to read the next event in the eventBuffer.
332 newEvent = parent -> readEvent();
333
334 if( newEvent.type == Shellminator::SHELL_EVENT_RESIZE ){
335 redraw = true;
336 return;
337 }
338
339}

351void ShellminatorProgress::setPercentage( float percentage_p ){
352 if( parent == NULL ){
353 return;
354 }
355
356 if( ( (int)( percentage_p * 10.0 ) ) != ( (int)( percentage * 10.0 ) ) ){
357 percentage = percentage_p;
358 redraw = true;
359 parent -> requestRedraw();
360 }
