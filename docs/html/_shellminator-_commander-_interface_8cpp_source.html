100void ShellminatorCommanderInterface::autoCompleteWithCommandParser(){
101
102 int num_of_hints;
103 int i;
104 const char* hint_text;
105 int hint_text_size;
106
107 // By default it is empty, because no command parser is available.
108 if( commander == NULL ){
109 return;
110 }
111
112 if( autoCompleteBuffer == NULL ){
113 return;
114 }
115
116 if( autoCompleteBufferSize < 10 ){
117 return;
118 }
119
120 if( cmd_buff_cntr >= SHELLMINATOR_BUFF_LEN ){
121 return;
122 }
123
124 cmd_buff[ 0 ][ cmd_buff_cntr ] = '\0';
125 num_of_hints = commander -> generateHint( cmd_buff[ 0 ], autoCompleteBuffer, autoCompleteBufferSize );
126
127
128 if( num_of_hints < 1 ){
129 return;
130 }
131
132 hint_text = commander -> getHint( 0, true );
133 hint_text_size = strlen( hint_text );
134
135 for( i = 0; i < hint_text_size; i++ ){
136 cmd_buff[ 0 ][ cmd_buff_cntr ] = hint_text[ i ];
137 cmd_buff_cntr++;
138 cursor++;
139 if( cmd_buff_cntr >= SHELLMINATOR_BUFF_LEN ){
140 break;
141 }
142 }
143
144 redrawLine();
145
146}

165void CommanderColorizer::reset( Stream* response ){
166 int i;
167
168 for( i = 0; i < ( SHELLMINATOR_BUFF_LEN + 1 ); i++ ){
169 stateBuffer[ i ] = DEFAULT_STATE;
170 }
171
172 bufferCntr = 0;
173 currentState = DEFAULT_STATE;
174
175 if( response ){
176 response -> print( "\033[0m" );

180void CommanderColorizer::printChar( Stream* response, char c ){
181
182 if( bufferCntr > SHELLMINATOR_BUFF_LEN ){
183 return;
184 }
185
186 switch( currentState ){
187
188 case DEFAULT_STATE:
189 defaultStateFuncion( response, c );
190 break;
191
192 case FIRST_DASH_STATE:
193 firstDashStateFuncion( response, c );
194 break;
195
196 case WAIT_FOR_WHITESPACE:
197 waitForWhitespaceStateFuncion( response, c );
198 break;
199
200 case SECOND_DASH_STATE:
201 secondDashStateFuncion( response, c );
202 break;
203
204 case WAIT_TOKEN_END:
205 waitTokenEndStateFunction( response, c );
206 break;
207
208 case WAIT_STRING_END:
209 waitStringEndStateFunction( response, c );
210 break;
211
212 case ENV_VAR_START:
213 envVarStartStateFunction( response, c );
214 break;
215
216
217 }
218
219 bufferCntr++;
220 stateBuffer[ bufferCntr ] = currentState;
221
222}

224void CommanderColorizer::defaultStateFuncion( Stream* response, char c ){
225
226 /*
227 if( ( ( c >= 'A' ) && ( c <= 'Z' ) ) || ( ( c >= 'a' ) && ( c <= 'z' ) ) || ( ( c >= '0' ) && ( c <= '9' ) ) ){
228 response -> print( c );
229 }
230 */
231
232 if( ( ( c == ' ' ) || ( c == '\t' ) ) && bufferCntr == 0 ){
233 response -> print( "\033[1;37;41m" );
234 response -> print( "\u00b7" );
235 response -> print( "\033[0m" );
236 return;
237 }
238
239 if( c == '-' ){
240 response -> print( "\033[1;35m" );
241 response -> print( c );
242 currentState = FIRST_DASH_STATE;
243 return;
244 }
245
246 if( c == '\"' ){
247 response -> print( "\033[1;36m" );
248 response -> print( c );
249 currentState = WAIT_STRING_END;
250 return;
251 }
252
253 if( c == '$' ){
254 response -> print( "\033[1;33m" );
255 response -> print( c );
256 currentState = ENV_VAR_START;
257 return;
258 }
259
260 response -> print( c );
261
262}

264void CommanderColorizer::firstDashStateFuncion( Stream* response, char c ){
265 if( c == '-' ){
266 response -> print( c );
267 currentState = SECOND_DASH_STATE;
268 return;
269 }
270
271 if( isCharacter( c ) || isNumber( c ) ){
272 response -> print( c );
273 currentState = WAIT_FOR_WHITESPACE;
274 return;
275 }
276
277 response -> print( "\033[1;37;41m" );
278 response -> print( c );
279 response -> print( "\033[0m" );
280 currentState = DEFAULT_STATE;
281
