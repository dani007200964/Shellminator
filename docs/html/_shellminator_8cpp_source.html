69 // Check if the buffer or its size is defined incorrectly.

163 return;

207void Shellminator::printHistory(){
208
209 // Generic counter.
210 uint32_t i;
211
212 // The index of the first valid command
213 // in the history.
214 uint32_t firstBuffDim;
215
216 // This variable will hold the actual
217 // history element order.
218 // 1 is the oldest.
219 uint32_t index;
220
221 // The first history element usually located in the
222 // 1-st index.
223 firstBuffDim = 1;
224
225 // Detect if the channel was configured incorrectly.
226 // In this case, we have to stop command execution.
227 if( channel == NULL ){
228 return;
229 }
230
231 // This will be redirected to the channel by default.
232 Stream* selectedChannel = channel;
233
234 // If buffering is enabled, overwrite the
235 // selectedChannel to the buffered printer.
236 if( bufferMemoryAllocated ){
237 selectedChannel = &bufferedPrinter;
238 }
239
240 // Check the first valid command in the buffer
241 for( i = 1; i < SHELLMINATOR_BUFF_DIM; i++ ){
242
243 // If it's found, store it's index to firstBuffDim
244 if( cmd_buff[ i ][ 0 ] == '\0' ){
245 firstBuffDim = i - 1;
246 break;
247 }
248
249 }
250
251 // If the history is empty, we can return.
252 if( firstBuffDim == 0 ){
253 return;
254 }
255
256 // If the history is full, we have to protect
257 // firstBuffDim variable.
258 if( i >= SHELLMINATOR_BUFF_DIM ){
259 firstBuffDim = SHELLMINATOR_BUFF_DIM - 1;
260 }
261
262 // Print the history.
263 for( i = firstBuffDim; i > 0; i-- ){
264
265 index = firstBuffDim - i + 1;
266
267 selectedChannel -> print( ' ' );
268 selectedChannel -> print( ' ' );
269
270 // It is used to ident digits.
271 if( index < 10 ){
272
273 selectedChannel -> print( ' ' );
274 selectedChannel -> print( ' ' );
275
276 }
277
278 // It is used to ident digits.
279 else if( index < 100 ){
280
281 selectedChannel -> print( ' ' );
282
283 }
284
285 // Print the index and the command.
286 format( selectedChannel, BOLD, MAGENTA );
287 selectedChannel -> print( index );
288 format( selectedChannel, REGULAR, WHITE );
289 selectedChannel -> print( ' ' );
290 selectedChannel -> print( ' ' );
291 selectedChannel -> println( cmd_buff[ i ] );
292
293 }
294
295 // Check if buffering is enabled.
296 // If so, we have to flush the buffer to the output stream.
297 if( bufferMemoryAllocated ){
298 selectedChannel -> flush();
299 }

437void Shellminator::begin( const char* banner_p ) {
438
439 // If channel is not specified, we have to stop here.
440 if( channel == NULL ){
441 return;
442 }
443
444 // Copy the content from banner_p to banner. Because strncpy we can be sure that it wont overflow.
445 strncpy( banner, banner_p, SHELLMINATOR_BANNER_LEN );
446
447 // Just in case close the string
448 banner[ SHELLMINATOR_BANNER_LEN - 1 ] = '\0';
449
450 // Set the terminal color and style to the defined settings for the logo
451 format( channel, SHELLMINATOR_LOGO_FONT_STYLE, SHELLMINATOR_LOGO_COLOR );
452
453 // Draw the startup logo.
454 drawLogo();
455
456 // By default set the login status to true.
457 // It means password is not required.
458 loggedIn = true;
459
460 // Check if the password hash buffer is not NULL.
461 // It means, that a password hash is attached and
462 // login is required to use the terminal.
463 if( passwordHash != NULL ){
464
465 // Clear the loggedIn flag to force the user
466 // to enter the correct password.
467 loggedIn = false;
468
469 // Create a special input prompt for internal use.
470 // Because it is used internally, we did not specify
471 // an output buffer and did nos specify a callback.
472 // The ShellminatorEnterKeyState function will identify
473 // this prompt by the loggedIn flag.
474 input( NULL, SHELLMINATOR_BUFF_LEN, "Password:", NULL, true );
475
476 }
477
478 else{
479
480 // If we have an empty string as banner, it probably means that we need
481 // an initialised shell, but we still configuring it. It thats the case,
482 // we skip the banner printing here.
483 if( banner[ 0 ] == '\0' ){
484 return;
485 }
486
487 // Print the banner message.
488 printBanner();
489
490 }

506void Shellminator::redrawLine(){
507
508 // -- Note --
509 //
510 // Even if formatting is disabled, this function requires VT100 commands.
511 // Please not use it if you dont't want any formatting.
512
513 // Generic counter.
514 uint32_t i;
515
516 // General counter variable
517 int j = -1;
518
519 // Detect if the channel was configured incorrectly.
520 // In this case, we have to stop command execution.
521 if( channel == NULL ){
522 return;
523 }
524
525 // This will be redirected to the channel by default.
526 Stream* selectedChannel= channel;
527
528 // If buffering is enabled, overwrite the
529 // selectedChannel to the buffered printer.
530 if( bufferMemoryAllocated ){
531 selectedChannel = &bufferedPrinter;
532 }
533
534 colorizer -> reset( selectedChannel );
535
536 // If reverse search mode is active,
537 // we have to call redrawHistorySearch
538 // function and than stop there.
539 if( inSearch ){
540
541 redrawHistorySearch();
542
543 // Check if buffering is enabled.
544 // If so, we have to flush the buffer to the output stream.
545 if( bufferMemoryAllocated ){
546 selectedChannel -> flush();
547 }
548
549 return;
550
551 }
552
553 // Check if it is out of boundaries and
554 // limit the variable to a maximum allowed value
555 // to prevent buffer overflow.
556 if( cmd_buff_cntr > SHELLMINATOR_BUFF_LEN ){
557
558 cmd_buff_cntr = SHELLMINATOR_BUFF_LEN;
559
560 }
561
562 // Terminate the command at the cmd_buff_cntr
563 // to not print out the previous command's data.
564 cmd_buff[ 0 ][ cmd_buff_cntr ] = '\0';
565
566 // Return to the beginning of the line and print the banner
567 // then the command buffer will be printed (with colors)
568 selectedChannel -> print( '\r' );
569
570 // We have to print the banner only, when no input
571 // prompt is active.
572 if( !inputActive ){
573 printBanner();
574 }
575
576 // If an input prompt is active, we have to set the cursor
577 // to the end of the input prompts instruction text.
578 else{
579 selectedChannel -> write( 27 ); // ESC character( decimal 27 )
580 selectedChannel -> print( '[' ); // VT100 Cursor command.
581 selectedChannel -> print( inputInstuctionSize ); // Step cursor
582 selectedChannel -> print( 'C' ); // Left.
583 }
584
585 // If an input is not active, we print the user text according to
586 // the colorizers output.
587 if( !inputActive ){
588 colorizer -> reset( selectedChannel );
589 for( i = 0; i < cmd_buff_cntr; i++ ){
590 colorizer -> printChar( selectedChannel, cmd_buff[ 0 ][ i ] );
591 }
592 }
593
594 // If we an active input, we just print the characters without a colorizer.
595 // If the input is in secret mode, we replace the characters with a dot character.
596 else{
597 for( i = 0; i < cmd_buff_cntr; i++ ){
598 if( inputSecretMode ){
599 selectedChannel -> print( "\u2022" );
600 }
601 else{
602 selectedChannel -> print( cmd_buff[ i ] );
603 }
604 }
605 }
606
607 // Clear the rest of the line to right.
608 selectedChannel -> print( __CONST_TXT__( "\033[0m\033[0K" ) );
609
610 // It is also a tricky part. We have to check if the cursor
611 // is not at the end of the command. In this case, we have
612 // to set back the cursor to its intended position.
613 if( cmd_buff_cntr > cursor ){
614
615 selectedChannel -> write( 27 ); // ESC character( decimal 27 )
616 selectedChannel -> print( '[' ); // VT100 Cursor command.
617 selectedChannel -> print( uint8_t( cmd_buff_cntr - cursor ) ); // Step cursor
618 selectedChannel -> print( 'D' ); // Left.
619
620 }
621
622 // Restore the color and style for user text.
623 format( selectedChannel, REGULAR, WHITE );
624
625 // Check if buffering is enabled.
626 // If so, we have to flush the buffer to the output stream.
627 if( bufferMemoryAllocated ){
628 selectedChannel -> flush();
629 }
630
631}

640void Shellminator::update() {
641
642 // This variable will hold the character that was read from the channel buffer.
643 char c;
644
645 // In case of screen drawing, the terminal width
646 // will be queried directly to this variable.
647 int w;
648
649 // In case of screen drawing, the terminal width
650 // will be queried directly to this variable.
651 int h;
652
653 // If channel is not specified, we have to stop here.
654 if( channel == NULL ){
655 return;
656 }
657
658 // We have to check the channel buffer. If it is not empty we can read as many characters as possible.
659 while ( channel -> available() ) {
660
661 // Read one character from channel Buffer.
662 c = (char)channel -> read();
663
664 // Process the new character.
665 process( c );
666
667 // Save the system time when the last keypress was executed.
668 lastKeyPressTime = millis();
669
670 }
671
672 // Check if a Screen object is attached.
673 if( screen != NULL ){
674
675 // If a Screen is attached, we have to request the terminal
676 // size periodically. The intervall for this is specidied
677 // in the screenUpdatePeriod variable.
678 if( ( millis() - sizeTimerStart ) > 2000 ){
679
680 // Save system time to detect the next timing event.
681 sizeTimerStart = millis();
682
683 // Try to get the screen size.
684 getTerminalSize( &w, &h );
685
686 // Check if the width or the height has been changed since the
687 // last iteration.
688 if( ( terminalWidth != w ) || ( terminalHeight != h ) ){
689
690 // If the terminal size has been changed, we have to signal the
691 // attached Screen. Maybe the Screen object has to redraw itself
692 // to adopt for the new screen size.
693 pushEvent( ( shellEvent_t ){ SHELL_EVENT_RESIZE } );
694
695 // Call the update function of the screen with the new
696 // dimensions to make it possible to adopt to the new
697 // screen size.
698 screen -> update( w, h );
699
700 // Set the redraw flag to true.
701 screenRedraw = true;
702
703 }
704
705 // Save the dimensions to internal variables.
706 terminalWidth = w;
707 terminalHeight = h;
708
709 }
710
711 if( screenRedraw && ( ( millis() - screenTimerStart ) > screenUpdatePeriod ) ){
712
713 screenTimerStart = millis();
714 // In this case we have to clear the flag first.
715 // The order is important, because it is possible
716 // that the Screen object will generate a redraw
717 // request in the draw function.
718 screenRedraw = false;
719
720 screen -> forceRedraw();
721
722 // Call the Screen objects draw function.
723 screen -> draw();
724
725 // Check if buffering is enabled.
726 // If so, we have to flush the buffer to the output stream.
727 if( bufferMemoryAllocated ){
728 bufferedPrinter.flush();
729 }
730 }
731
732 // We have to check screen again, because an endScreen was called
733 // in the draw callback, it results a crash.
734 if( screen ){
735 // Call the Screen objects update function periodically.
736 screen -> update( terminalWidth, terminalHeight );
737 }
738
739 // Pop the current element from the event buffer.
740 popEvent();
741
742 }
743
744 checkCommandFraction();
745
746}

748bool Shellminator::getCursorPosition( int* x, int* y, uint32_t timeout ){
749
750 // Screen coordinate report looks like this:
751 // ESC[yCoord;xCoordR
752 // ESC[10;3R
753
754 // It will hold the current character.
755 char c;
756
757 // Stores the current state of the state-machine.
758 // Default state is 0.
759 uint8_t charState = 0;
760
761 // Buffer for the X-coordinate text.
762 char xBuff[ 4 ];
763
764 // This variable tracks the index of the
765 // next free slot in the xBuff array.
766 uint8_t xBuffCntr = 0;
767
768 // Buffer for the Y-coordinate text.
769 char yBuff[ 4 ];
770
771 // This variable tracks the index of the
772 // next free slot in the yBuff array.
773 uint8_t yBuffCntr = 0;
774
775 // To save system time for timeout detection.
776 unsigned long timerStart;
777
778 // If channel is not specified, we have to stop here.
779 if( channel == NULL ){
780 return false;
781 }
782
783 // Flush the input buffer.
784 while( channel -> available() ){
785 channel -> read();
786 }
787
788 // Send the request.
789 channel -> print( __CONST_TXT__( "\033[6n" ) );
790
791 // Save system time.
792 timerStart = millis();
793
794 // If no timeout event, or no timeout specified, check for new data.
795 while( ( ( millis() - timerStart ) < timeout ) || timeout == 0 ){
796
797 // Process all data if available.
798 while( channel -> available() ){
799
800 // Read character
801 c = channel -> read();
802
803 switch( charState ){
804
805 // Inital state. We expect ESC character.
806 case 0:
807
808 // If ESC character is received, go to the next state.
809 if( c == 27 ){
810 charState = 1;
811 }
812
813 // Otherwise, something wrong.
814 else{
815 //channel -> println( "\r\nERROR: CASE 0\r\n" );
816 return false;
817 }
818
819 break;
820
821 // Second character state. We expect '[' character.
822 case 1:
823
824 // If '[' character is received, go to the next state.
825 if( c == '[' ){
826 charState = 2;
827 }
828
829 // Otherwise, something wrong.
830 else{
831 //channel -> println( "\r\nERROR: CASE 1\r\n" );
832 return false;
833 }
834
835 break;
836
837 // We wait for ';' character. Until this character we put the data in
838 // the yBuff;
839 case 2:
840
841 // If ';' character is received, go to the next state.
842 if( c == ';' ){
843 charState = 3;
844 yBuff[ yBuffCntr ] = '\0';
845 }
846
847 else{
848
849 yBuff[ yBuffCntr ] = c;
850 yBuffCntr++;
851
852 // Protection against buffer overflow.
853 if( yBuffCntr >= 4 ){
854 //channel -> println( "\r\nERROR: CASE 2\r\n" );
855 return false;
856 }
857
858 }
859
860 break;
861
862 // We wait for 'R' character. Until this character we put the data in
863 // the yBuff;
864 case 3:
865
866 // If 'R' character is received, we finished with data collecting.
867 if( c == 'R' ){
868 xBuff[ xBuffCntr ] = '\0';
869
870 // We have to parse the numbers.
871 charState = sscanf( (const char*)xBuff, "%d", x );
872 charState += sscanf( (const char*)yBuff, "%d", y );
873
874 // Check if all data is number.
875 if( charState == 2 ){
876 return true;
877 }
878
879 // Conversion went wrong.
880 else{
881 //channel -> println( "\r\nERROR: CASE 3 R\r\n" );
882 return false;
883 }
884
885 }
886
887 else{
888
889 xBuff[ xBuffCntr ] = c;
890 xBuffCntr++;
891
892 // Protection against buffer overflow.
893 if( xBuffCntr >= 4 ){
894 //channel -> println( "\r\nERROR: CASE 3\r\n" );
895 return false;
896 }
897
898 }
899
900 break;
901
902 }
903
904 }
905
906 // If emscripten is teh build target, a small sleep is necessary.
907 // The xterm js frontend can do its job, while the 'c++' code sleeps.
908 #ifdef EMSCRIPTEN
909 emscripten_sleep( 20 );
910 #endif
911
912 }
913
914 //channel -> println( "\r\nTIMEOUT!\r\n" );
915
916 // We should not be here.
917 return false;
918

1232
1233 }
1234

1238 // rework the buffer and redraw the whole line
1239 for( i = cursor; i < cmd_buff_cntr; i++ ) {

1253void Shellminator::ShellminatorEnterKeyState(){
1254
1255 // If the enter key is pressed cmd_buff_dim has to be reset to the default value
1256 cmd_buff_dim = 1;
1257
1258 // General counter variable
1259 int i;
1260
1261 // If channel is not specified, we have to stop here.
1262 if( channel == NULL ){
1263 return;
1264 }
1265
1266 colorizer -> reset( channel );
1267
1268 if( inSearch ){
1269
1270 if( searchMatch > 0 ){
1271
1272 inSearch = false;
1273 strncpy( cmd_buff[ 0 ], cmd_buff[ searchMatch ], SHELLMINATOR_BUFF_LEN + 1 );
1274 cmd_buff_cntr = strlen( cmd_buff[ 0 ] );
1275 redrawLine();
1276
1277 }
1278
1279 else{
1280 cmd_buff_cntr = 0;
1281 }
1282
1283 }
1284
1285 // Because a command is sent we have to close it. Basically we replace the arrived
1286 // '\r' character with a '\0' string terminator character. Now we have our command
1287 // in a C/C++ like standard string format.
1288 cmd_buff[ 0 ][ cmd_buff_cntr ] = '\0';
1289
1290 // Check if we have an active input prompt.
1291 if( inputActive ){
1292
1293 // Check if we are waiting for user password.
1294 if( !loggedIn ){
1295
1296 // To empty the incoming string we have to zero it's counter.
1297 cmd_buff_cntr = 0;
1298 cursor = 0;
1299
1300 // Check if the user typed the correct password.
1301 if( checkPassword( cmd_buff[ 0 ] ) ){
1302 loggedIn = true;
1303 inputActive = false;
1304 channel -> println();
1305 printBanner();
1306 return;
1307 }
1308
1309 // Otherwise generate a beep sound and start over the
1310 // password section.
1311 beep();
1312 redrawLine();
1313 return;
1314
1315 }
1316
1317 // If we are waiting for regular text input.
1318 else{
1319
1320 inputActive = false;
1321
1322 // Copy the data from the input to the destination.
1323 strncpy( inputDestinationBuffer, cmd_buff[ 0 ], inputDestinationBufferSize );
1324
1325 // Just in case terminate the string.
1326 inputDestinationBuffer[ inputDestinationBufferSize - 1 ] = '\0';
1327
1328 // To empty the incoming string we have to zero it's counter.
1329 cmd_buff_cntr = 0;
1330 cursor = 0;
1331
1332 if( inputCallback != NULL ){
1333 inputCallback( inputDestinationBuffer, inputDestinationBufferSize, this );
1334 }
1335
1336 // If the callback created a new input, we need to return here.
1337 if( inputActive ){
1338 return;
1339 }
1340
1341 // If we doesn't has any more input prompts, close the prompt,
1342 // and return to regular operation.
1343 channel -> println();
1344 printBanner();
1345 return;
1346
1347 }
1348
1349 }
1350
1351 // We send a line break to the terminal to put the next data in new line
1352 channel -> println();
1353
1354 inSearch = false;
1355
1356 // If the arrived data is not just a single enter we have to process the command.
1357 if ( cmd_buff_cntr > 0 ) {
1358
1359 // Flush out everything before command execution.
1360 // It is required for the prompts, because sometimes garbage
1361 // can be found in the channel stream. It has to be removed before
1362 // prompt reading.
1363 while( channel -> available() ){
1364 channel -> read();
1365 }
1366
1367 if( ( strcmp( cmd_buff[ 0 ], "help" ) == 0 ) || ( strcmp( cmd_buff[ 0 ], "?" ) == 0 ) ){
1368 printHelp();
1369 }
1370
1371 else if( strcmp( cmd_buff[ 0 ], "history" ) == 0 ){
1372 printHistory();
1373 }
1374
1375 else if( strcmp( cmd_buff[ 0 ], "neofetch" ) == 0 ){
1376 if( neofetch_fn != NULL ){
1377 neofetch_fn( this );
1378 }
1379 }
1380
1381 // We have to check that execution_fn is not NULL.
1382 else if( execution_fn != NULL ){
1383 // If it is a valid, then call it's function.
1384 execution_fn( cmd_buff[ 0 ], this );
1385 }
1386
1387 else if( hasCommandParser() ){
1388 executeWithCommandParser();
1389 }
1390
1391 // If not, then just print it with Serial.
1392 else{
1393 channel -> print( __CONST_TXT__( "cmd: " ) );
1394 channel -> print( cmd_buff[ 0 ] );
1395 channel -> println();
1396 }
1397
1398 // After we processed the command we have to shift the history upwards.
1399 // To protect the copy against buffer overflow we use strncpy
1400 for ( i = ( SHELLMINATOR_BUFF_DIM - 1 ); i > 0; i-- ) {
1401 strncpy( cmd_buff[ i ], cmd_buff[ i - 1 ], SHELLMINATOR_BUFF_LEN + 1 );
1402 }
1403
1404 }
1405
1406 // After the command processing finished we print a new banner to the terminal.
1407 // This means that the device is finished and waits the new command.
1408 printBanner();
1409
1410 // To empty the incoming string we have to zero it's counter.
1411 cmd_buff_cntr = 0;
1412 cursor = 0;
1413
1414}

1693bool Shellminator::waitForKey( Stream* source, char key, uint32_t timeout ){
1694
1695 // Save system time
1696 unsigned long timerStart = millis();
1697
1698 while( source -> available() < 1 ){
1699 // Wait for key input.
1700
1701 // Check if timeout is specified.
1702 if( timeout == 0 ){
1703 continue;
1704 }
1705
1706 // Timeout detection.
1707 if( ( millis() - timerStart ) > timeout ){
1708 return false;
1709 }
1710
1711 }
1712
