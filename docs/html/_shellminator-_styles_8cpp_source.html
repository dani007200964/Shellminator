188void Shellminator::clear() {
189
190 // explanation can be found here: http://braun-home.net/michael/info/misc/VT100_commands.htm
191 channel -> write( 27 ); // ESC character( decimal 27 )
192 channel -> print( (const char*)"[H" ); // VT100 Home command
193 channel -> write( 27 ); // ESC character( decimal 27 )
194 channel -> print( (const char*)"[J" ); // VT100 screen erase command
195
196}
