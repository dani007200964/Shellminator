46void ShellminatorNotification::setText( const char* text_p ){
47 int i;
48 int textWidthMax = 0;
49
50 text = text_p;
51
52 textWidth = 0;
53 textHeight = 1;
54
55 for( i = 0; i < strlen( text ); i++ ){
56
57 if( text[ i ] == '\n' ){
58 textHeight++;
59 textWidth = 0;
60 continue;
61 }
62
63 textWidth++;
64 if( textWidth > textWidthMax ){
65 textWidthMax = textWidth;
66 }
67
68 }
69
70 textWidth = textWidthMax;
71}

79void ShellminatorNotification::update( int width_p, int height_p ){
80
81 Shellminator::shellEvent_t newEvent;
82
83 if( parent == NULL ){
84 return;
85 }
86
87 width = width_p;
88 height = height_p;
89
90 newEvent = parent -> readEvent();
91
92 // In case of empty event, we can't do anything.
93 if( newEvent.type == Shellminator::SHELL_EVENT_EMPTY ){
94 return;
95 }
96
97 if( newEvent.type == Shellminator::SHELL_EVENT_RESIZE ){
98 redraw = true;
99 return;
100 }
101
102}

104void ShellminatorNotification::draw( bool noClear ){
105
106 // Generic counter.
107 int i;
108 int x;
109 int y;
110 int widthCounter;
111 int heightCounter;
112
113 int borderWidth;
114 int borderHeight;
115
116 int xStart;
117 int yStart;
118
119 // Check for the necessary parameters.
120 if( parent == NULL ){
121 return;
122 }
123
124 if( channel == NULL){
125 return;
126 }
127
128 if( text == NULL ){
129 return;
130 }
131
132 // Only draw if resized event or timer event happened.
133 if( !redraw ){
134 return;
135 }
136 redraw = false;
137
138 borderWidth = textWidth + 4;
139 borderHeight = textHeight + 2;
140
141 if( borderWidth > width ){
142 borderWidth = width;
143 }
144
145 if( borderHeight > height ){
146 borderHeight = height;
147 }
148
149 xStart = originX + width / 2 - borderWidth / 2;
150 yStart = originY + height / 2 - borderHeight / 2;
151
152 parent -> format( channel, Shellminator::REGULAR, Shellminator::WHITE );
153
154 // Set cursor to top left and print the instruction text.
155 Shellminator::setCursorPosition( channel, xStart, yStart );
156
157 channel -> print( __CONST_TXT__( "\u250C" ) );
158
159 for( i = 0; i < borderWidth - 2; i++ ){
160 channel -> print( __CONST_TXT__( "\u2500" ) );
161 }
162
163 channel -> print( __CONST_TXT__( "\u2510" ) );
164
165
166 i = 0;
167 for( y = 0; y < borderHeight - 2; y++ ){
168 Shellminator::setCursorPosition( channel, xStart, yStart + y + 1 );
169 channel -> print( __CONST_TXT__( "\u2502 " ) );
170 for( x = 0; x < borderWidth - 2; x++ ){
171 if( text[ i ] == '\n' ){
172 i++;
173 break;
174 }
175 if( text[ i ] == '\0' ){
176 break;
177 }
178 channel -> print( text[ i ] );
179 i++;
180 }
181 Shellminator::setCursorPosition( channel, xStart + borderWidth - 2, yStart + y + 1 );
182 channel -> print( __CONST_TXT__( " \u2502" ) );
183 }
184
185 // Set cursor to top left and print the instruction text.
186 Shellminator::setCursorPosition( channel, xStart, yStart + borderHeight - 1 );
187
188 channel -> print( __CONST_TXT__( "\u2514" ) );
189
190 for( i = 0; i < borderWidth - 2; i++ ){
191 channel -> print( __CONST_TXT__( "\u2500" ) );
192 }
193
194 channel -> print( __CONST_TXT__( "\u2139" ) );
195
196}
